```go
package types

import (
	"crypto/ecdsa"
	"math/big"
	"time"

	"github.com/ethereum/go-ethereum/common"
	"github.com/ethereum/go-ethereum/crypto"
	"github.com/ethereum/go-ethereum/rlp"
)

// TxStatus represents the lifecycle state of a transaction
type TxStatus string

const (
	TxPending    TxStatus = "PENDING"
	TxValidated  TxStatus = "VALIDATED"
	TxInMempool  TxStatus = "IN_MEMPOOL"
	TxSettled    TxStatus = "SETTLED"
	TxFailed     TxStatus = "FAILED"
	TxReversed   TxStatus = "REVERSED"
)

// Transaction is the canonical transaction model
type Transaction struct {
	// Core fields
	Nonce    uint64   `json:"nonce"`
	From     common.Address `json:"from"`
	To       *common.Address `json:"to"`
	Value    *big.Int `json:"value"`
	GasLimit uint64   `json:"gasLimit"`
	GasPrice *big.Int `json:"gasPrice"`
	Data     []byte   `json:"data"`

	// Signature fields (EIP-155)
	V *big.Int `json:"v" rlp:"nil"`
	R *big.Int `json:"r" rlp:"nil"`
	S *big.Int `json:"s" rlp:"nil"`

	// Derived fields (not serialized, computed on validation)
	Hash      common.Hash    `json:"hash" rlp:"-"`
	Sender    common.Address `json:"sender" rlp:"-"`
	Size      int            `json:"size" rlp:"-"`
	Timestamp time.Time      `json:"timestamp" rlp:"-"`

	// NexaPay extensions
	IdempotencyKey string            `json:"idempotencyKey" rlp:"-"`
	Metadata       map[string]string `json:"metadata" rlp:"-"`
	Status         TxStatus          `json:"status" rlp:"-"`
}

// HashRLP returns the keccak256 hash of the RLP-encoded transaction
func (tx *Transaction) HashRLP() (common.Hash, error) {
	rlpBytes, err := rlp.EncodeToBytes(tx)
	if err != nil {
		return common.Hash{}, err
	}
	return crypto.Keccak256Hash(rlpBytes), nil
}

// Sign signs the transaction with the given private key
func (tx *Transaction) Sign(privateKey *ecdsa.PrivateKey, chainID *big.Int) error {
	signer := crypto.NewEIP155Signer(chainID)
	txHash := signer.Hash(tx)

	signature, err := crypto.Sign(txHash.Bytes(), privateKey)
	if err != nil {
		return err
	}

	tx.V = new(big.Int).SetBytes(signature[64:])
	tx.R = new(big.Int).SetBytes(signature[0:32])
	tx.S = new(big.Int).SetBytes(signature[32:64])

	return nil
}

// RecoverSender recovers the sender address from the signature
func (tx *Transaction) RecoverSender(chainID *big.Int) (common.Address, error) {
	signer := crypto.NewEIP155Signer(chainID)
	return signer.Sender(tx)
}

// Validate performs structural validation
func (tx *Transaction) Validate() error {
	if tx.Value == nil || tx.Value.Sign() < 0 {
		return ErrInvalidValue
	}
	if tx.GasLimit == 0 {
		return ErrInvalidGasLimit
	}
	if tx.GasPrice == nil || tx.GasPrice.Sign() <= 0 {
		return ErrInvalidGasPrice
	}
	if tx.Nonce > 1e12 {
		return ErrInvalidNonce
	}
	return nil
}
```

```go
package crypto

import (
	"crypto/ecdsa"
	"crypto/rand"
	"errors"

	"github.com/ethereum/go-ethereum/common"
	"github.com/ethereum/go-ethereum/crypto"
)

var (
	ErrInvalidSignature = errors.New("invalid signature")
	ErrKeyGeneration    = errors.New("key generation failed")
)

// Signer handles transaction signing and verification
type Signer struct {
	privateKey *ecdsa.PrivateKey
	address    common.Address
}

// NewSigner creates a signer from a private key
func NewSigner(privateKeyHex string) (*Signer, error) {
	privateKey, err := crypto.HexToECDSA(privateKeyHex)
	if err != nil {
		return nil, err
	}

	publicKey := privateKey.Public()
	publicKeyECDSA, ok := publicKey.(*ecdsa.PublicKey)
	if !ok {
		return nil, ErrKeyGeneration
	}

	address := crypto.PubkeyToAddress(*publicKeyECDSA)

	return &Signer{
		privateKey: privateKey,
		address:    address,
	}, nil
}

// GenerateSigner creates a new random signer
func GenerateSigner() (*Signer, error) {
	privateKey, err := ecdsa.GenerateKey(crypto.S256(), rand.Reader)
	if err != nil {
		return nil, err
	}

	publicKey := privateKey.Public()
	publicKeyECDSA := publicKey.(*ecdsa.PublicKey)
	address := crypto.PubkeyToAddress(*publicKeyECDSA)

	return &Signer{
		privateKey: privateKey,
		address:    address,
	}, nil
}

// Address returns the signer's address
func (s *Signer) Address() common.Address {
	return s.address
}

// SignHash signs a keccak256 hash
func (s *Signer) SignHash(hash []byte) ([]byte, error) {
	signature, err := crypto.Sign(hash, s.privateKey)
	if err != nil {
		return nil, err
	}
	// Adjust V value for Ethereum compatibility
	signature[64] += 27
	return signature, nil
}

// VerifySignature verifies a signature against an address
func VerifySignature(address common.Address, hash []byte, signature []byte) bool {
	if len(signature) != 65 {
		return false
	}

	// Adjust V value back
	signature[64] -= 27

	pubKey, err := crypto.SigToPub(hash, signature)
	if err != nil {
		return false
	}

	recoveredAddr := crypto.PubkeyToAddress(*pubKey)
	return recoveredAddr == address
}

// Keccak256 computes the keccak256 hash of data
func Keccak256(data []byte) []byte {
	return crypto.Keccak256(data)
}

// Keccak256Hash computes the keccak256 hash and returns as Hash
func Keccak256Hash(data []byte) common.Hash {
	return crypto.Keccak256Hash(data)
}
```

```go
package mempool

import (
	"container/heap"
	"sync"
	"time"

	"nexapay/internal/types"
)

// Pool is a production mempool with per-account nonce tracking
type Pool struct {
	mu sync.RWMutex

	// Global priority queue (by gas price)
	globalQueue PriorityQueue

	// Per-account nonce-ordered queues
	accountQueues map[string]*AccountQueue

	// Nonce tracking
	nonceTracker *NonceTracker

	// Eviction engine
	eviction *EvictionEngine

	// Configuration
	maxSize      int
	maxPerAccount int
	minGasPrice  uint64

	// Metrics
	totalTxs     int64
	evictedTxs   int64
	replacedTxs  int64
}

// NewPool creates a production mempool
func NewPool(maxSize, maxPerAccount int, minGasPrice uint64) *Pool {
	p := &Pool{
		globalQueue:   make(PriorityQueue, 0),
		accountQueues: make(map[string]*AccountQueue),
		nonceTracker:  NewNonceTracker(),
		maxSize:       maxSize,
		maxPerAccount: maxPerAccount,
		minGasPrice:   minGasPrice,
	}
	p.eviction = NewEvictionEngine(p)
	heap.Init(&p.globalQueue)
	return p
}

// Add attempts to add a transaction to the mempool
func (p *Pool) Add(tx *types.Transaction) error {
	p.mu.Lock()
	defer p.mu.Unlock()

	// Validate gas price
	if tx.GasPrice.Uint64() < p.minGasPrice {
		return ErrGasPriceTooLow
	}

	// Enforce size limit
	if p.Size() >= p.maxSize {
		p.eviction.EvictLowest()
	}

	addr := tx.From.Hex()
	aq, exists := p.accountQueues[addr]
	if !exists {
		aq = NewAccountQueue(addr, p.maxPerAccount)
		p.accountQueues[addr] = aq
	}

	// Check for replacement (same nonce, higher gas)
	if existing := aq.GetByNonce(tx.Nonce); existing != nil {
		if tx.GasPrice.Cmp(existing.GasPrice) <= 0 {
			return ErrNonceAlreadyUsed
		}
		// Replace with higher gas price
		aq.Replace(existing, tx)
		p.replacedTxs++
		return nil
	}

	// Add to account queue
	if err := aq.Add(tx); err != nil {
		return err
	}

	// Add to global priority queue
	item := &PriorityItem{
		tx:        tx,
		gasPrice:  tx.GasPrice.Uint64(),
		timestamp: time.Now(),
	}
	heap.Push(&p.globalQueue, item)
	p.totalTxs++

	return nil
}

// GetNext returns the highest-priority transaction
func (p *Pool) GetNext() *types.Transaction {
	p.mu.Lock()
	defer p.mu.Unlock()

	if p.globalQueue.Len() == 0 {
		return nil
	}

	item := heap.Pop(&p.globalQueue).(*PriorityItem)
	return item.tx
}

// Size returns the number of transactions in the pool
func (p *Pool) Size() int {
	p.mu.RLock()
	defer p.mu.RUnlock()
	return p.globalQueue.Len()
}

// Remove removes a transaction from the pool
func (p *Pool) Remove(hash string) {
	p.mu.Lock()
	defer p.mu.Unlock()

	for i, item := range p.globalQueue {
		if item.tx.Hash.Hex() == hash {
			heap.Remove(&p.globalQueue, i)
			addr := item.tx.From.Hex()
			if aq, exists := p.accountQueues[addr]; exists {
				aq.Remove(item.tx.Nonce)
			}
			return
		}
	}
}

// FlushAccount removes all transactions for an account
func (p *Pool) FlushAccount(addr string) {
	p.mu.Lock()
	defer p.mu.Unlock()

	delete(p.accountQueues, addr)

	// Rebuild global queue without this account's transactions
	newQueue := make(PriorityQueue, 0)
	for _, item := range p.globalQueue {
		if item.tx.From.Hex() != addr {
			newQueue = append(newQueue, item)
		}
	}
	p.globalQueue = newQueue
	heap.Init(&p.globalQueue)
}

// Stats returns mempool statistics
func (p *Pool) Stats() MempoolStats {
	p.mu.RLock()
	defer p.mu.RUnlock()

	return MempoolStats{
		TotalTxs:    p.totalTxs,
		PendingTxs:  p.globalQueue.Len(),
		EvictedTxs:  p.evictedTxs,
		ReplacedTxs: p.replacedTxs,
		Accounts:    len(p.accountQueues),
	}
}
```

```go
package state

import (
	"database/sql"
	"errors"
	"time"

	"github.com/google/uuid"
)

var (
	ErrInsufficientFunds = errors.New("insufficient funds")
	ErrAccountNotFound   = errors.New("account not found")
	ErrIdempotencyFail   = errors.New("idempotency key already used")
)

// Ledger is a double-entry accounting system
type Ledger struct {
	db *sql.DB
}

// Entry represents a single ledger entry (immutable)
type Entry struct {
	ID            string    `json:"id"`
	AccountID     string    `json:"accountId"`
	Type          string    `json:"type"` // CREDIT or DEBIT
	Amount        float64   `json:"amount"`
	EntryType     string    `json:"entryType"`
	ReferenceID   string    `json:"referenceId"`
	IdempotencyKey string   `json:"idempotencyKey"`
	CreatedAt     time.Time `json:"createdAt"`
}

// Account represents a ledger account
type Account struct {
	ID               string    `json:"id"`
	UserID           string    `json:"userId"`
	Currency         string    `json:"currency"`
	Balance          float64   `json:"balance"`
	AvailableBalance float64   `json:"availableBalance"`
	Version          int       `json:"version"`
	CreatedAt        time.Time `json:"createdAt"`
}

// PostTransaction posts a double-entry transaction
func (l *Ledger) PostTransaction(params PostTxParams) (*TransactionResult, error) {
	tx, err := l.db.Begin()
	if err != nil {
		return nil, err
	}
	defer tx.Rollback()

	// Idempotency check
	var existingID string
	err = tx.QueryRow(
		"SELECT id FROM entries WHERE idempotency_key = $1 LIMIT 1",
		params.IdempotencyKey,
	).Scan(&existingID)
	if err == nil {
		return &TransactionResult{Status: "DUPLICATE", ID: existingID}, nil
	}

	// Lock sender account
	var sender Account
	err = tx.QueryRow(
		"SELECT id, balance, available_balance, version FROM accounts WHERE id = $1 FOR UPDATE",
		params.SenderAccountID,
	).Scan(&sender.ID, &sender.Balance, &sender.AvailableBalance, &sender.Version)
	if err != nil {
		return nil, ErrAccountNotFound
	}

	// Check sufficient funds
	if sender.AvailableBalance < params.Amount {
		return &TransactionResult{Status: "FAILED", Reason: "INSUFFICIENT_FUNDS"}, nil
	}

	// Create transaction record
	txnID := uuid.New().String()
	_, err = tx.Exec(`
		INSERT INTO transactions (id, sender_account_id, receiver_account_id, amount, currency, status, idempotency_key, description)
		VALUES ($1, $2, $3, $4, $5, 'SETTLED', $6, $7)`,
		txnID, params.SenderAccountID, params.ReceiverAccountID,
		params.Amount, params.Currency, params.IdempotencyKey, params.Description,
	)
	if err != nil {
		return nil, err
	}

	// Post debit entry
	_, err = tx.Exec(`
		INSERT INTO entries (account_id, type, amount, entry_type, reference_id, idempotency_key)
		VALUES ($1, 'DEBIT', $2, 'TRANSFER_SEND', $3, $4)`,
		params.SenderAccountID, params.Amount, txnID, params.IdempotencyKey+"_debit",
	)
	if err != nil {
		return nil, err
	}

	// Post credit entry
	_, err = tx.Exec(`
		INSERT INTO entries (account_id, type, amount, entry_type, reference_id, idempotency_key)
		VALUES ($1, 'CREDIT', $2, 'TRANSFER_RECEIVE', $3, $4)`,
		params.ReceiverAccountID, params.Amount, txnID, params.IdempotencyKey+"_credit",
	)
	if err != nil {
		return nil, err
	}

	// Update balances
	_, err = tx.Exec(
		"UPDATE accounts SET balance = balance - $1, available_balance = available_balance - $1, version = version + 1 WHERE id = $2",
		params.Amount, params.SenderAccountID,
	)
	if err != nil {
		return nil, err
	}

	_, err = tx.Exec(
		"UPDATE accounts SET balance = balance + $1, available_balance = available_balance + $1, version = version + 1 WHERE id = $2",
		params.Amount, params.ReceiverAccountID,
	)
	if err != nil {
		return nil, err
	}

	if err = tx.Commit(); err != nil {
		return nil, err
	}

	return &TransactionResult{
		ID:     txnID,
		Status: "SETTLED",
	}, nil
}
```

```go
package middleware

import (
	"crypto/sha256"
	"encoding/hex"
	"sync"
	"time"
)

// ReplayProtection prevents transaction replay attacks
type ReplayProtection struct {
	mu       sync.RWMutex
	seen     map[string]time.Time
	maxAge   time.Duration
	maxSize  int
}

// NewReplayProtection creates a replay protection instance
func NewReplayProtection(maxAge time.Duration, maxSize int) *ReplayProtection {
	rp := &ReplayProtection{
		seen:    make(map[string]time.Time),
		maxAge:  maxAge,
		maxSize: maxSize,
	}
	go rp.cleanup()
	return rp
}

// Check returns true if the request is a replay
func (rp *ReplayProtection) Check(idempotencyKey string) bool {
	rp.mu.RLock()
	_, exists := rp.seen[idempotencyKey]
	rp.mu.RUnlock()
	return exists
}

// Mark records a request as seen
func (rp *ReplayProtection) Mark(idempotencyKey string) {
	rp.mu.Lock()
	defer rp.mu.Unlock()

	if len(rp.seen) >= rp.maxSize {
		// Evict oldest entry
		var oldestKey string
		var oldestTime time.Time
		for k, t := range rp.seen {
			if oldestKey == "" || t.Before(oldestTime) {
				oldestKey = k
				oldestTime = t
			}
		}
		delete(rp.seen, oldestKey)
	}

	rp.seen[idempotencyKey] = time.Now()
}

// cleanup removes expired entries
func (rp *ReplayProtection) cleanup() {
	ticker := time.NewTicker(time.Minute)
	for range ticker.C {
		rp.mu.Lock()
		cutoff := time.Now().Add(-rp.maxAge)
		for k, t := range rp.seen {
			if t.Before(cutoff) {
				delete(rp.seen, k)
			}
		}
		rp.mu.Unlock()
	}
}

// HashRequest creates a hash of the request for deduplication
func HashRequest(method string, params []byte) string {
	h := sha256.New()
	h.Write([]byte(method))
	h.Write(params)
	return hex.EncodeToString(h.Sum(nil))
}
```
