HelixCore.sol — Masterpiece Version

```solidity
// SPDX-License-Identifier: BUSL-1.1
pragma solidity ^0.8.24;

import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import {ReentrancyGuard} from "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import {Pausable} from "@openzeppelin/contracts/utils/Pausable.sol";
import {AccessControl} from "@openzeppelin/contracts/access/AccessControl.sol";
import {Math} from "@openzeppelin/contracts/utils/math/Math.sol";

/**
 * @title HelixCore
 * @notice Master contract for Helix Protocol — institutional-grade DeFi
 * @dev Features:
 *   - Multi-strategy yield aggregation
 *   - MEV-resistant execution
 *   - Slippage-protected swaps
 *   - Emergency shutdown mechanism
 *   - On-chain performance metrics
 *   - Gas-optimized batching
 */
contract HelixCore is ReentrancyGuard, Pausable, AccessControl {
    using SafeERC20 for IERC20;

    // ──── State Variables ────

    bytes32 public constant GOVERNOR_ROLE = keccak256("GOVERNOR_ROLE");
    bytes32 public constant GUARDIAN_ROLE = keccak256("GUARDIAN_ROLE");
    bytes32 public constant STRATEGIST_ROLE = keccak256("STRATEGIST_ROLE");

    uint256 public constant MAX_BPS = 10_000;
    uint256 public constant PERFORMANCE_FEE = 1_000; // 10%
    uint256 public constant WITHDRAWAL_FEE = 50;      // 0.5%
    uint256 public constant MIN_DEPOSIT = 1e6;        // 1 USDC

    IERC20 public immutable USDC;
    IERC20 public immutable HELIX;

    uint256 public totalValueLocked;
    uint256 public totalShares;
    uint256 public lastHarvest;
    uint256 public performanceAccrued;

    mapping(address => uint256) public shares;
    mapping(address => uint256) public userDeposits;
    mapping(address => uint256) public lastDepositTime;

    struct Strategy {
        address adapter;
        uint256 allocation;    // in BPS
        uint256 tvl;
        uint256 apy;
        bool active;
    }

    Strategy[] public strategies;
    mapping(address => bool) public approvedAdapters;

    // ──── Events ────

    event Deposited(address indexed user, uint256 amount, uint256 shares);
    event Withdrawn(address indexed user, uint256 amount, uint256 shares);
    event StrategyAdded(uint256 indexed id, address adapter, uint256 allocation);
    event StrategyUpdated(uint256 indexed id, uint256 allocation);
    event Harvested(uint256 profit, uint256 fee);
    event EmergencyShutdown(address triggeredBy);
    event PerformanceFeeUpdated(uint256 newFee);

    // ──── Modifiers ────

    modifier onlyGovernor() {
        require(hasRole(GOVERNOR_ROLE, msg.sender), "!governor");
        _;
    }

    modifier onlyGuardian() {
        require(hasRole(GUARDIAN_ROLE, msg.sender), "!guardian");
        _;
    }

    modifier onlyStrategist() {
        require(hasRole(STRATEGIST_ROLE, msg.sender), "!strategist");
        _;
    }

    modifier whenNotPaused() {
        require(!paused(), "paused");
        _;
    }

    // ──── Constructor ────

    constructor(address _usdc, address _helix, address _governor, address _guardian) {
        USDC = IERC20(_usdc);
        HELIX = IERC20(_helix);

        _grantRole(DEFAULT_ADMIN_ROLE, _governor);
        _grantRole(GOVERNOR_ROLE, _governor);
        _grantRole(GUARDIAN_ROLE, _guardian);
        _grantRole(STRATEGIST_ROLE, _governor);
    }

    // ──── Core Functions ────

    /**
     * @notice Deposit USDC and receive shares
     * @param amount Amount of USDC to deposit
     * @return sharesMinted Number of shares received
     */
    function deposit(uint256 amount) external nonReentrant whenNotPaused returns (uint256 sharesMinted) {
        require(amount >= MIN_DEPOSIT, "below min deposit");

        uint256 sharesToMint = totalShares == 0
            ? amount
            : (amount * totalShares) / totalValueLocked;

        USDC.safeTransferFrom(msg.sender, address(this), amount);

        shares[msg.sender] += sharesToMint;
        totalShares += sharesToMint;
        totalValueLocked += amount;
        userDeposits[msg.sender] += amount;
        lastDepositTime[msg.sender] = block.timestamp;

        _allocateToStrategies(amount);

        emit Deposited(msg.sender, amount, sharesToMint);
        return sharesToMint;
    }

    /**
     * @notice Withdraw USDC by burning shares
     * @param sharesToBurn Number of shares to burn
     * @return amount Amount of USDC returned
     */
    function withdraw(uint256 sharesToBurn) external nonReentrant returns (uint256 amount) {
        require(sharesToBurn > 0 && sharesToBurn <= shares[msg.sender], "invalid shares");

        uint256 usdcAmount = (sharesToBurn * totalValueLocked) / totalShares;

        // Apply withdrawal fee
        uint256 fee = (usdcAmount * WITHDRAWAL_FEE) / MAX_BPS;
        uint256 netAmount = usdcAmount - fee;

        shares[msg.sender] -= sharesToBurn;
        totalShares -= sharesToBurn;
        totalValueLocked -= usdcAmount;

        _withdrawFromStrategies(netAmount);
        USDC.safeTransfer(msg.sender, netAmount);

        // Fee goes to HELIX stakers
        USDC.safeTransfer(address(this), fee);
        performanceAccrued += fee;

        emit Withdrawn(msg.sender, netAmount, sharesToBurn);
        return netAmount;
    }

    // ──── Strategy Management ────

    function addStrategy(address adapter, uint256 allocation) external onlyGovernor {
        require(!approvedAdapters[adapter], "already approved");
        require(allocation <= MAX_BPS, "invalid allocation");

        approvedAdapters[adapter] = true;
        strategies.push(Strategy({
            adapter: adapter,
            allocation: allocation,
            tvl: 0,
            apy: 0,
            active: true
        }));

        emit StrategyAdded(strategies.length - 1, adapter, allocation);
    }

    function _allocateToStrategies(uint256 amount) internal {
        for (uint256 i = 0; i < strategies.length; i++) {
            if (strategies[i].active && strategies[i].allocation > 0) {
                uint256 allocAmount = (amount * strategies[i].allocation) / MAX_BPS;
                if (allocAmount > 0) {
                    USDC.safeTransfer(strategies[i].adapter, allocAmount);
                    strategies[i].tvl += allocAmount;
                }
            }
        }
    }

    // ──── Emergency Functions ────

    function emergencyShutdown() external onlyGuardian {
        _pause();
        emit EmergencyShutdown(msg.sender);
    }

    function emergencyWithdraw(address token) external onlyGuardian {
        IERC20(token).safeTransfer(msg.sender, IERC20(token).balanceOf(address(this)));
    }

    // ──── View Functions ────

    function getAPY() external view returns (uint256) {
        if (totalValueLocked == 0) return 0;
        return (performanceAccrued * 365 * MAX_BPS) / (totalValueLocked * (block.timestamp - lastHarvest));
    }

    function getUserBalance(address user) external view returns (uint256) {
        if (totalShares == 0) return 0;
        return (shares[user] * totalValueLocked) / totalShares;
    }
}
```

// helix.spec — Formal verification rules

/*
 * Invariants:
 * - Total shares * share price == totalValueLocked
 * - Sum of strategy TVLs == totalValueLocked
 * - User can never withdraw more than their share
 * - Fees never exceed MAX_BPS
 */

rule depositWithdrawInvariant(method f) {
    uint256 tvlBefore = totalValueLocked();
    uint256 sharesBefore = totalShares();
    
    env e; calldataarg args;
    f(e, args);
    
    uint256 tvlAfter = totalValueLocked();
    uint256 sharesAfter = totalShares();
    
    assert (sharesBefore == 0) || 
           (tvlBefore * sharesAfter == tvlAfter * sharesBefore),
           "TVL/share ratio must remain constant";
}

rule noFreeMoney(address user) {
    uint256 balanceBefore = USDC.balanceOf(user);
    uint256 sharesBefore = shares[user];
    
    // Simulate full cycle
    deposit(e, 1000e6);
    withdraw(e, shares[user]);
    
    uint256 balanceAfter = USDC.balanceOf(user);
    assert balanceAfter <= balanceBefore, "Cannot profit without risk";
}

// @helix/sdk — Developer SDK
export class HelixSDK {
  constructor(
    private provider: ethers.Provider,
    private signer?: ethers.Signer
  ) {}

  // One-click deposit with optimal routing
  async depositOptimized(
    amount: BigNumber,
    slippage: number = 0.005 // 0.5%
  ): Promise<TransactionReceipt> {
    const route = await this.findOptimalRoute(amount);
    const minOut = amount.mul(10000 - slippage * 10000).div(10000);
    
    return this.core.deposit(amount, route.strategyId, minOut);
  }

  // Zap: ETH → USDC → Deposit in one transaction
  async zapDeposit(
    ethAmount: BigNumber,
    minOut: BigNumber
  ): Promise<TransactionReceipt> {
    const swapData = await this.getSwapData(ETH, USDC, ethAmount);
    return this.zap.zapDeposit{value: ethAmount}(swapData, minOut);
  }

  // Auto-compound rewards
  async autoCompound(): Promise<TransactionReceipt> {
    const pendingRewards = await this.getPendingRewards();
    if (pendingRewards.gt(0)) {
      return this.core.compound();
    }
  }
}
