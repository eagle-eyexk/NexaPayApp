```rust
// services/ledger-engine/src/ledger/entry.rs

use serde::{Deserialize, Serialize};
use sha2::{Sha256, Digest};
use chrono::{DateTime, Utc};
use uuid::Uuid;

/// LedgerEntry is the atomic unit of financial truth
/// Once written, it is NEVER modified or deleted
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct LedgerEntry {
    /// Unique entry ID (UUIDv7 for time-based ordering)
    pub entry_id: String,
    
    /// Transaction this entry belongs to
    pub transaction_id: String,
    
    /// Wallet affected
    pub wallet_id: String,
    
    /// DEBIT or CREDIT
    pub entry_type: EntryType,
    
    /// Amount in NexaToken smallest unit (like satoshis)
    pub amount: u64,
    
    /// Balance BEFORE this entry (for verification)
    pub balance_before: u64,
    
    /// Balance AFTER this entry
    pub balance_after: u64,
    
    /// Human-readable description
    pub description: String,
    
    /// Arbitrary metadata
    pub metadata: serde_json::Value,
    
    /// When this entry was created
    pub created_at: DateTime<Utc>,
    
    /// Sequence number in the ledger (monotonically increasing)
    pub sequence_number: u64,
    
    /// Hash of the previous entry (forms a chain)
    pub previous_entry_hash: String,
    
    /// Hash of this entry
    pub entry_hash: String,
}

#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
pub enum EntryType {
    DEBIT,
    CREDIT,
}

impl LedgerEntry {
    /// Create a new ledger entry
    pub fn new(
        transaction_id: String,
        wallet_id: String,
        entry_type: EntryType,
        amount: u64,
        balance_before: u64,
        description: String,
        previous_entry_hash: String,
        sequence_number: u64,
    ) -> Self {
        let entry_id = Uuid::now_v7().to_string();
        let balance_after = match entry_type {
            EntryType::CREDIT => balance_before.saturating_add(amount),
            EntryType::DEBIT => balance_before.saturating_sub(amount),
        };
        
        let mut entry = Self {
            entry_id: entry_id.clone(),
            transaction_id,
            wallet_id,
            entry_type,
            amount,
            balance_before,
            balance_after,
            description,
            metadata: serde_json::Value::Object(serde_json::Map::new()),
            created_at: Utc::now(),
            sequence_number,
            previous_entry_hash,
            entry_hash: String::new(),
        };
        
        // Compute hash after creation
        entry.entry_hash = entry.compute_hash();
        
        entry
    }
    
    /// Compute SHA-256 hash of the entry
    pub fn compute_hash(&self) -> String {
        let mut hasher = Sha256::new();
        
        hasher.update(self.entry_id.as_bytes());
        hasher.update(self.transaction_id.as_bytes());
        hasher.update(self.wallet_id.as_bytes());
        hasher.update(match self.entry_type {
            EntryType::DEBIT => b"DEBIT",
            EntryType::CREDIT => b"CREDIT",
        });
        hasher.update(self.amount.to_le_bytes());
        hasher.update(self.balance_before.to_le_bytes());
        hasher.update(self.balance_after.to_le_bytes());
        hasher.update(self.previous_entry_hash.as_bytes());
        hasher.update(self.sequence_number.to_le_bytes());
        hasher.update(self.created_at.timestamp_micros().to_le_bytes());
        
        hex::encode(hasher.finalize())
    }
    
    /// Verify the entry is self-consistent
    pub fn verify(&self) -> Result<(), String> {
        // Verify balance calculation
        let expected_balance_after = match self.entry_type {
            EntryType::CREDIT => self.balance_before.checked_add(self.amount),
            EntryType::DEBIT => self.balance_before.checked_sub(self.amount),
        };
        
        match expected_balance_after {
            Some(expected) if expected == self.balance_after => {},
            Some(_) => return Err(format!(
                "Balance mismatch: expected {} but got {}",
                expected_balance_after.unwrap(),
                self.balance_after
            )),
            None => return Err("Balance overflow/underflow".to_string()),
        }
        
        // Verify hash
        let computed_hash = self.compute_hash();
        if computed_hash != self.entry_hash {
            return Err("Hash mismatch".to_string());
        }
        
        Ok(())
    }
}
```

```rust
// services/ledger-engine/src/ledger/journal.rs

use std::collections::HashMap;
use std::sync::{Arc, RwLock};
use tokio::sync::Mutex;
use crate::ledger::entry::{LedgerEntry, EntryType};

/// The Journal is the append-only ledger
/// All financial truth derives from this
pub struct Journal {
    /// All entries in sequence order
    entries: Vec<LedgerEntry>,
    
    /// Index: wallet_id -> latest balance
    wallet_balances: HashMap<String, u64>,
    
    /// Index: transaction_id -> list of entries
    transaction_index: HashMap<String, Vec<usize>>,
    
    /// Current sequence number
    sequence: u64,
    
    /// Hash of the last entry (chain integrity)
    last_hash: String,
    
    /// Total supply tracking
    total_debits: u64,
    total_credits: u64,
}

impl Journal {
    pub fn new() -> Self {
        Self {
            entries: Vec::new(),
            wallet_balances: HashMap::new(),
            transaction_index: HashMap::new(),
            sequence: 0,
            last_hash: String::from("GENESIS"),
            total_debits: 0,
            total_credits: 0,
        }
    }
    
    /// Append a new entry to the journal
    pub fn append(&mut self, mut entry: LedgerEntry) -> Result<&LedgerEntry, String> {
        // Verify the entry
        entry.verify()?;
        
        // Verify sequence number
        if entry.sequence_number != self.sequence + 1 {
            return Err(format!(
                "Invalid sequence: expected {}, got {}",
                self.sequence + 1,
                entry.sequence_number
            ));
        }
        
        // Verify previous hash
        if entry.previous_entry_hash != self.last_hash {
            return Err(format!(
                "Chain broken: expected previous hash {}, got {}",
                self.last_hash,
                entry.previous_entry_hash
            ));
        }
        
        // Get current wallet balance
        let current_balance = self.wallet_balances
            .get(&entry.wallet_id)
            .copied()
            .unwrap_or(0);
        
        // Verify balance_before matches
        if entry.balance_before != current_balance {
            return Err(format!(
                "Balance before mismatch: expected {}, got {}",
                current_balance,
                entry.balance_before
            ));
        }
        
        // Update wallet balance
        self.wallet_balances.insert(
            entry.wallet_id.clone(),
            entry.balance_after,
        );
        
        // Update totals
        match entry.entry_type {
            EntryType::CREDIT => self.total_credits += entry.amount,
            EntryType::DEBIT => self.total_debits += entry.amount,
        }
        
        // Store entry index
        let entry_index = self.entries.len();
        self.transaction_index
            .entry(entry.transaction_id.clone())
            .or_insert_with(Vec::new)
            .push(entry_index);
        
        // Update chain state
        self.sequence = entry.sequence_number;
        self.last_hash = entry.entry_hash.clone();
        
        // Append to journal
        self.entries.push(entry);
        
        Ok(&self.entries[entry_index])
    }
    
    /// Get wallet balance (derived from entries)
    pub fn get_balance(&self, wallet_id: &str) -> u64 {
        self.wallet_balances.get(wallet_id).copied().unwrap_or(0)
    }
    
    /// Get all entries for a wallet
    pub fn get_wallet_entries(&self, wallet_id: &str) -> Vec<&LedgerEntry> {
        self.entries
            .iter()
            .filter(|e| e.wallet_id == wallet_id)
            .collect()
    }
    
    /// Get entries for a transaction
    pub fn get_transaction_entries(&self, transaction_id: &str) -> Vec<&LedgerEntry> {
        self.transaction_index
            .get(transaction_id)
            .map(|indices| {
                indices.iter().filter_map(|&i| self.entries.get(i)).collect()
            })
            .unwrap_or_default()
    }
    
    /// Verify global zero-sum invariant
    /// Total credits must equal total debits
    pub fn verify_zero_sum(&self) -> Result<(), String> {
        if self.total_credits != self.total_debits {
            return Err(format!(
                "Zero-sum violation: credits={}, debits={}, diff={}",
                self.total_credits,
                self.total_debits,
                self.total_credits.abs_diff(self.total_debits)
            ));
        }
        Ok(())
    }
    
    /// Get total supply (all credits - all debits should be zero in a closed system)
    pub fn get_total_supply(&self) -> u64 {
        // In a closed system, this should represent minted tokens
        // For NexaToken, this is managed by the token engine
        self.total_credits
    }
    
    /// Get total entry count
    pub fn entry_count(&self) -> usize {
        self.entries.len()
    }
    
    /// Get current sequence number
    pub fn current_sequence(&self) -> u64 {
        self.sequence
    }
    
    /// Reconstruct balances from scratch (for recovery)
    pub fn reconstruct_balances(&mut self) {
        self.wallet_balances.clear();
        self.total_debits = 0;
        self.total_credits = 0;
        
        for entry in &self.entries {
            self.wallet_balances.insert(
                entry.wallet_id.clone(),
                entry.balance_after,
            );
            
            match entry.entry_type {
                EntryType::CREDIT => self.total_credits += entry.amount,
                EntryType::DEBIT => self.total_debits += entry.amount,
            }
        }
    }
}
```

```rust
// services/ledger-engine/src/consensus/raft.rs

use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::{Mutex, RwLock};
use serde::{Deserialize, Serialize};

/// Raft consensus implementation for distributed ledger
/// Ensures all nodes agree on entry order

#[derive(Debug, Clone, PartialEq)]
pub enum NodeState {
    Follower,
    Candidate,
    Leader,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct LogEntry {
    pub term: u64,
    pub index: u64,
    pub command: Vec<u8>, // Serialized ledger entry
}

pub struct RaftNode {
    id: String,
    state: NodeState,
    current_term: u64,
    voted_for: Option<String>,
    log: Vec<LogEntry>,
    commit_index: u64,
    last_applied: u64,
    next_index: HashMap<String, u64>,
    match_index: HashMap<String, u64>,
    peers: Vec<String>,
    election_timeout: u64,
    heartbeat_interval: u64,
}

impl RaftNode {
    pub fn new(id: String, peers: Vec<String>) -> Self {
        Self {
            id,
            state: NodeState::Follower,
            current_term: 0,
            voted_for: None,
            log: Vec::new(),
            commit_index: 0,
            last_applied: 0,
            next_index: HashMap::new(),
            match_index: HashMap::new(),
            peers,
            election_timeout: 150, // milliseconds
            heartbeat_interval: 50, // milliseconds
        }
    }
    
    /// Start an election
    pub fn start_election(&mut self) {
        self.state = NodeState::Candidate;
        self.current_term += 1;
        self.voted_for = Some(self.id.clone());
        
        // In production: send RequestVote RPCs to all peers
        println!("Node {} starting election for term {}", self.id, self.current_term);
    }
    
    /// Become leader
    pub fn become_leader(&mut self) {
        self.state = NodeState::Leader;
        
        // Initialize next_index and match_index for all peers
        let last_log_index = self.log.len() as u64;
        for peer in &self.peers {
            self.next_index.insert(peer.clone(), last_log_index + 1);
            self.match_index.insert(peer.clone(), 0);
        }
        
        println!("Node {} became leader for term {}", self.id, self.current_term);
    }
    
    /// Append entry to log (leader only)
    pub fn append_entry(&mut self, command: Vec<u8>) -> Result<u64, String> {
        if self.state != NodeState::Leader {
            return Err("Not the leader".to_string());
        }
        
        let entry = LogEntry {
            term: self.current_term,
            index: self.log.len() as u64 + 1,
            command,
        };
        
        self.log.push(entry);
        
        // In production: replicate to followers
        
        Ok(self.log.len() as u64)
    }
    
    /// Commit entries up to index
    pub fn commit_entries(&mut self, up_to_index: u64) {
        if up_to_index > self.commit_index {
            self.commit_index = up_to_index;
        }
    }
    
    /// Get committed entries since last_applied
    pub fn get_committed_entries(&self) -> Vec<&LogEntry> {
        self.log
            .iter()
            .filter(|e| e.index > self.last_applied && e.index <= self.commit_index)
            .collect()
    }
}
```

```rust
// services/token-engine/src/token/transfer.rs

use serde::{Deserialize, Serialize};
use sha2::{Sha256, Digest};
use ed25519_dalek::{Signer, Verifier, Signature, SigningKey, VerifyingKey};
use crate::token::balance::TokenBalance;

/// NexaToken Transfer
/// The fundamental unit of value movement
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TokenTransfer {
    /// Unique transfer ID
    pub transfer_id: String,
    
    /// Sender's wallet address (public key)
    pub sender: String,
    
    /// Receiver's wallet address (public key)
    pub receiver: String,
    
    /// Amount in NexaToken base units
    pub amount: u64,
    
    /// Fee paid for this transfer
    pub fee: u64,
    
    /// Nonce from sender (prevents replay)
    pub nonce: u64,
    
    /// When this transfer was created
    pub timestamp: u64,
    
    /// Transfer metadata
    pub metadata: Option<String>,
    
    /// Cryptographic signature
    pub signature: Vec<u8>,
}

impl TokenTransfer {
    /// Create a new unsigned transfer
    pub fn new(
        sender: String,
        receiver: String,
        amount: u64,
        fee: u64,
        nonce: u64,
        metadata: Option<String>,
    ) -> Self {
        Self {
            transfer_id: uuid::Uuid::new_v4().to_string(),
            sender,
            receiver,
            amount,
            fee,
            nonce,
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_secs(),
            metadata,
            signature: Vec::new(),
        }
    }
    
    /// Compute the hash of this transfer (for signing)
    pub fn compute_hash(&self) -> Vec<u8> {
        let mut hasher = Sha256::new();
        hasher.update(self.transfer_id.as_bytes());
        hasher.update(self.sender.as_bytes());
        hasher.update(self.receiver.as_bytes());
        hasher.update(&self.amount.to_le_bytes());
        hasher.update(&self.fee.to_le_bytes());
        hasher.update(&self.nonce.to_le_bytes());
        hasher.update(&self.timestamp.to_le_bytes());
        if let Some(ref meta) = self.metadata {
            hasher.update(meta.as_bytes());
        }
        hasher.finalize().to_vec()
    }
    
    /// Sign the transfer with sender's private key
    pub fn sign(&mut self, signing_key: &SigningKey) {
        let hash = self.compute_hash();
        let signature = signing_key.sign(&hash);
        self.signature = signature.to_bytes().to_vec();
    }
    
    /// Verify the transfer signature
    pub fn verify(&self, verifying_key: &VerifyingKey) -> Result<(), String> {
        if self.signature.is_empty() {
            return Err("Transfer is not signed".to_string());
        }
        
        let hash = self.compute_hash();
        let signature = Signature::from_slice(&self.signature)
            .map_err(|e| format!("Invalid signature bytes: {}", e))?;
        
        verifying_key
            .verify(&hash, &signature)
            .map_err(|e| format!("Signature verification failed: {}", e))
    }
    
    /// Validate the transfer (business rules)
    pub fn validate(
        &self,
        sender_balance: &TokenBalance,
    ) -> Result<(), String> {
        // Check amount > 0
        if self.amount == 0 {
            return Err("Transfer amount must be greater than zero".to_string());
        }
        
        // Check sufficient balance (amount + fee)
        let total_cost = self.amount.checked_add(self.fee)
            .ok_or("Amount overflow")?;
        
        if sender_balance.available < total_cost {
            return Err(format!(
                "Insufficient balance: available={}, required={}",
                sender_balance.available,
                total_cost
            ));
        }
        
        // Check nonce (must be exactly sender's current nonce + 1)
        if self.nonce != sender_balance.nonce + 1 {
            return Err(format!(
                "Invalid nonce: expected {}, got {}",
                sender_balance.nonce + 1,
                self.nonce
            ));
        }
        
        // Check sender != receiver
        if self.sender == self.receiver {
            return Err("Cannot transfer to self".to_string());
        }
        
        Ok(())
    }
}

/// Anti-double-spend detection
pub struct DoubleSpendGuard {
    /// Set of processed transfer IDs
    processed_transfers: std::collections::HashSet<String>,
    
    /// Map of sender -> used nonces
    used_nonces: std::collections::HashMap<String, std::collections::HashSet<u64>>,
}

impl DoubleSpendGuard {
    pub fn new() -> Self {
        Self {
            processed_transfers: std::collections::HashSet::new(),
            used_nonces: std::collections::HashMap::new(),
        }
    }
    
    /// Check if transfer is a double-spend attempt
    pub fn check(&self, transfer: &TokenTransfer) -> Result<(), String> {
        // Check if transfer ID already processed
        if self.processed_transfers.contains(&transfer.transfer_id) {
            return Err(format!(
                "Transfer {} already processed",
                transfer.transfer_id
            ));
        }
        
        // Check if nonce already used for this sender
        if let Some(nonces) = self.used_nonces.get(&transfer.sender) {
            if nonces.contains(&transfer.nonce) {
                return Err(format!(
                    "Nonce {} already used for sender {}",
                    transfer.nonce,
                    transfer.sender
                ));
            }
        }
        
        Ok(())
    }
    
    /// Record a processed transfer
    pub fn record(&mut self, transfer: &TokenTransfer) {
        self.processed_transfers.insert(transfer.transfer_id.clone());
        self.used_nonces
            .entry(transfer.sender.clone())
            .or_insert_with(std::collections::HashSet::new)
            .insert(transfer.nonce);
    }
}
```

```rust
use serde::{Deserialize, Serialize};

/// Token supply management
/// Only the governance contract can mint or burn tokens
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TokenSupply {
    /// Total tokens ever minted
    pub total_minted: u64,
    
    /// Total tokens burned
    pub total_burned: u64,

    pub circulating_supply: u64,

    pub max_supply: u64,
    
    /// Governance address that can mint/burn
    pub governance_address: String,

    pub mint_events: Vec<MintEvent>,
    
    /// Burn events for audit

}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct MintEvent {
    pub event_id: String,
    pub recipient: String,
    pub amount: u64,
    pub reason: String,
    pub authorized_by: String,
    pub timestamp: u64,
    pub transaction_hash: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BurnEvent {
    pub event_id: String,
    pub from: String,
    pub amount: u64,
    pub reason: String,
    pub authorized_by: String,
    pub timestamp: u64,
    pub transaction_hash: String,
}

impl TokenSupply {
    pub fn new(initial_supply: u64, max_supply: u64, governance_address: String) -> Self {
        Self {
            total_minted: initial_supply,
            total_burned: 0,
            circulating_supply: initial_supply,
            max_supply,
            governance_address,
            mint_events: Vec::new(),
            burn_events: Vec::new(),
        }
    }
    
    /// Mint new tokens (governance only)
    pub fn mint(
        &mut self,
        recipient: String,
        amount: u64,
        reason: String,
        authorized_by: String,
    ) -> Result<MintEvent, String> {
        // Authorization check
        if authorized_by != self.governance_address {
            return Err("Only governance can mint tokens".to_string());
        }
        
        // Supply cap check
        if self.max_supply > 0 {
            let new_total = self.total_minted.checked_add(amount)
                .ok_or("Mint overflow")?;
            if new_total > self.max_supply {
                return Err(format!(
                    "Mint would exceed max supply: {} > {}",
                    new_total,
                    self.max_supply
                ));
            }
        }
        
        let event = MintEvent {
            event_id: uuid::Uuid::new_v4().to_string(),
            recipient,
            amount,
            reason,
            authorized_by,
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_secs(),
            transaction_hash: String::new(),
        };
        
        self.total_minted = self.total_minted.checked_add(amount)
            .ok_or("Mint overflow")?;
        self.circulating_supply = self.circulating_supply.checked_add(amount)
            .ok_or("Supply overflow")?;
        
        self.mint_events.push(event.clone());
        
        Ok(event)
    }
    
    /// Burn tokens (governance only)
    pub fn burn(
        &mut self,
        from: String,
        amount: u64,
        reason: String,
        authorized_by: String,
    ) -> Result<BurnEvent, String> {
        if authorized_by != self.governance_address {
            return Err("Only governance can burn tokens".to_string());
        }
        
        let event = BurnEvent {
            event_id: uuid::Uuid::new_v4().to_string(),
            from,
            amount,
            reason,
            authorized_by,
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_secs(),
            transaction_hash: String::new(),
        };
        
        self.total_burned = self.total_burned.checked_add(amount)
            .ok_or("Burn overflow")?;
        self.circulating_supply = self.circulating_supply.checked_sub(amount)
            .ok_or("Insufficient supply to burn")?;
        
        self.burn_events.push(event.clone());
        
        Ok(event)
    }
}
```

```go
// services/payment-engine/internal/payment/state_machine.go

package payment

import (
	"errors"
	"fmt"
	"sync"
	"time"

// PaymentState represents the state of a payment
type PaymentState string

const (
	StateCreated     PaymentState = "CREATED"
	StateAuthorized  PaymentState = "AUTHORIZED"
	StateCaptured    PaymentState = "CAPTURED"
	StateSettled     PaymentState = "SETTLED"
	StateFailed      PaymentState = "FAILED"
	StateReversed    PaymentState = "REVERSED"
	StateChargeback  PaymentState = "CHARGEBACK"
	StateExpired     PaymentState = "EXPIRED"
)

// PaymentAction represents an action that can be performed on a payment
type PaymentAction string

const (
	ActionAuthorize      PaymentAction = "AUTHORIZE"
	ActionCapture        PaymentAction = "CAPTURE"
	ActionSettle         PaymentAction = "SETTLE"
	ActionFail           PaymentAction = "FAIL"
	ActionReverse        PaymentAction = "REVERSE"
	ActionChargeback     PaymentAction = "CHARGEBACK"
	ActionExpire         PaymentAction = "EXPIRE"
	ActionPartialCapture PaymentAction = "PARTIAL_CAPTURE"
	ActionPartialRefund  PaymentAction = "PARTIAL_REFUND"
)

// ValidTransitions defines all valid state transitions
var ValidTransitions = map[PaymentState][]PaymentState{
	StateCreated:    {StateAuthorized, StateFailed, StateExpired},
	StateAuthorized: {StateCaptured, StateReversed, StateFailed, StateExpired},
	StateCaptured:   {StateSettled, StateReversed},
	StateSettled:    {StateChargeback},
	StateFailed:     {},
	StateReversed:   {},
	StateChargeback: {StateReversed},
	StateExpired:    {},
}

// ActionToTargetState maps actions to their target states
var ActionToTargetState = map[PaymentAction]PaymentState{
	ActionAuthorize:      StateAuthorized,
	ActionCapture:        StateCaptured,
	ActionSettle:         StateSettled,
	ActionFail:           StateFailed,
	ActionReverse:        StateReversed,
	ActionChargeback:     StateChargeback,
	ActionExpire:         StateExpired,
	ActionPartialCapture: StateCaptured,
	ActionPartialRefund:  StateReversed,
}

// StateTransition records a state change
type StateTransition struct {
	From        PaymentState `json:"from"`
	To          PaymentState `json:"to"`
	Action      PaymentAction `json:"action"`
	Timestamp   time.Time    `json:"timestamp"`
	PerformedBy string       `json:"performed_by"`
	Metadata    map[string]interface{} `json:"metadata,omitempty"`
}

// PaymentStateMachine manages payment lifecycle
type PaymentStateMachine struct {
	mu           sync.RWMutex
	paymentID    string
	currentState PaymentState
	history      []StateTransition
}

// NewPaymentStateMachine creates a new state machine
func NewPaymentStateMachine(paymentID string) *PaymentStateMachine {
	initialTransition := StateTransition{
		From:        StateCreated,
		To:          StateCreated,
		Action:      ActionAuthorize,
		Timestamp:   time.Now(),
		PerformedBy: "SYSTEM",
		Metadata:    map[string]interface{}{"reason": "Payment created"},
	}

	return &PaymentStateMachine{
		paymentID:    paymentID,
		currentState: StateCreated,
		history:      []StateTransition{initialTransition},
	}
}

// Transition attempts to transition to a new state
func (psm *PaymentStateMachine) Transition(
	action PaymentAction,
	performedBy string,
	metadata map[string]interface{},
) (*StateTransition, error) {
	psm.mu.Lock()
	defer psm.mu.Unlock()

	targetState, ok := ActionToTargetState[action]
	if !ok {
		return nil, fmt.Errorf("unknown action: %s", action)
	}

	// Check if transition is valid
	validTargets := ValidTransitions[psm.currentState]
	isValid := false
	for _, valid := range validTargets {
		if valid == targetState {
			isValid = true
			break
		}
	}

	if !isValid {
		return nil, fmt.Errorf(
			"invalid state transition: %s -> %s (valid: %v)",
			psm.currentState, targetState, validTargets,
		)
	}

	transition := StateTransition{
		From:        psm.currentState,
		To:          targetState,
		Action:      action,
		Timestamp:   time.Now(),
		PerformedBy: performedBy,
		Metadata:    metadata,
	}

	psm.currentState = targetState
	psm.history = append(psm.history, transition)

	return &transition, nil
}

// CanTransition checks if a transition is possible
func (psm *PaymentStateMachine) CanTransition(action PaymentAction) bool {
	psm.mu.RLock()
	defer psm.mu.RUnlock()

	targetState, ok := ActionToTargetState[action]
	if !ok {
		return false
	}

	for _, valid := range ValidTransitions[psm.currentState] {
		if valid == targetState {
			return true
		}
	}
	return false
}

// GetCurrentState returns the current state
func (psm *PaymentStateMachine) GetCurrentState() PaymentState {
	psm.mu.RLock()
	defer psm.mu.RUnlock()
	return psm.currentState
}

// GetHistory returns the full transition history
func (psm *PaymentStateMachine) GetHistory() []StateTransition {
	psm.mu.RLock()
	defer psm.mu.RUnlock()
	
	history := make([]StateTransition, len(psm.history))
	copy(history, psm.history)
	return history
}

// IsTerminal checks if the payment is in a terminal state
func (psm *PaymentStateMachine) IsTerminal() bool {
	psm.mu.RLock()
	defer psm.mu.RUnlock()
	
	return psm.currentState == StateFailed ||
		psm.currentState == StateReversed ||
		psm.currentState == StateExpired
}

// GetAvailableActions returns actions available in current state
func (psm *PaymentStateMachine) GetAvailableActions() []PaymentAction {
	psm.mu.RLock()
	defer psm.mu.RUnlock()
	
	validTargets := ValidTransitions[psm.currentState]
	var actions []PaymentAction
	
	for action, target := range ActionToTargetState {
		for _, valid := range validTargets {
			if valid == target {
				actions = append(actions, action)
				break
			}
		}
	}
	
	return actions
}

// ValidatePayment ensures payment follows all business rules
func ValidatePayment(amount uint64, currency string) error {
	if amount == 0 {
		return errors.New("payment amount must be greater than zero")
	}
	
	if currency != "NEXA" {
		return fmt.Errorf("unsupported currency: %s (only NEXA is supported)", currency)
	}
	
	if amount > 1000000000000 { // 1 trillion NEXA units
		return errors.New("payment amount exceeds maximum limit")
	}
	
	return nil
}
```

```typescript
// services/api-gateway/src/middleware/signing.ts

import { Request, Response, NextFunction } from 'express';
import * as crypto from 'crypto';

/**
 * Request Signing Middleware
 * 
 * Every mutating API request MUST include:
 * - X-Request-Timestamp: Unix timestamp in seconds
 * - X-Request-Signature: HMAC-SHA256(secret, timestamp + method + path + body)
 * 
 * This prevents:
 * - Replay attacks (timestamp window)
 * - Request tampering (signature verification)
 * - Man-in-the-middle attacks
 */

const SIGNING_SECRET = process.env.API_SIGNING_SECRET || '';
const REQUEST_WINDOW_SECONDS = 300; // 5 minutes
const MAX_BODY_SIZE = 10 * 1024 * 1024; // 10MB

interface SignedRequest extends Request {
  verifiedSignature?: boolean;
}

export function requestSigningMiddleware(
  req: SignedRequest,
  res: Response,
  next: NextFunction
): void {
  // Skip for GET requests (they don't mutate state)
  if (req.method === 'GET' || req.method === 'HEAD' || req.method === 'OPTIONS') {
    return next();
  }

  const timestamp = req.headers['x-request-timestamp'] as string;
  const signature = req.headers['x-request-signature'] as string;

  // Validate required headers
  if (!timestamp) {
    res.status(400).json({ error: 'X-Request-Timestamp header is required' });
    return;
  }

  if (!signature) {
    res.status(400).json({ error: 'X-Request-Signature header is required' });
    return;
  }

  // Anti-Replay: Validate timestamp
  const requestTime = parseInt(timestamp, 10);
  if (isNaN(requestTime)) {
    res.status(400).json({ error: 'Invalid timestamp format' });
    return;
  }

  const now = Math.floor(Date.now() / 1000);
  const timeDifference = Math.abs(now - requestTime);

  if (timeDifference > REQUEST_WINDOW_SECONDS) {
    res.status(401).json({
      error: `Request timestamp outside allowed window of ${REQUEST_WINDOW_SECONDS} seconds`,
      difference: timeDifference,
    });
    return;
  }

  // Validate signature
  if (!SIGNING_SECRET) {
    console.error('API_SIGNING_SECRET not configured');
    res.status(500).json({ error: 'Server configuration error' });
    return;
  }

  const bodyString = JSON.stringify(req.body || {});
  const payload = `${timestamp}:${req.method}:${req.path}:${bodyString}`;

  const expectedSignature = crypto
    .createHmac('sha256', SIGNING_SECRET)
    .update(payload)
    .digest('hex');

  // Constant-time comparison to prevent timing attacks
  try {
    if (!crypto.timingSafeEqual(
      Buffer.from(signature, 'hex'),
      Buffer.from(expectedSignature, 'hex')
    )) {
      res.status(401).json({ error: 'Invalid request signature' });
      return;
    }
  } catch (err) {
    res.status(401).json({ error: 'Invalid signature format' });
    return;
  }

  req.verifiedSignature = true;
  next();
}

/**
 * Generate a signature for client-side use
 */
export function generateSignature(
  secret: string,
  timestamp: number,
  method: string,
  path: string,
  body: object
): string {
  const bodyString = JSON.stringify(body);
  const payload = `${timestamp}:${method}:${path}:${bodyString}`;
  
  return crypto
    .createHmac('sha256', secret)
    .update(payload)
    .digest('hex');
}

/**
 * Create signed headers for API requests
 */
export function createSignedHeaders(
  secret: string,
  method: string,
  path: string,
  body: object
): Record<string, string> {
  const timestamp = Math.floor(Date.now() / 1000);
  const signature = generateSignature(secret, timestamp, method, path, body);
  
  return {
    'X-Request-Timestamp': timestamp.toString(),
    'X-Request-Signature': signature,
    'Content-Type': 'application/json',
  };
}
```

```typescript
// services/api-gateway/src/middleware/idempotency.ts

import { Request, Response, NextFunction } from 'express';
import Redis from 'ioredis';

/**
 * Idempotency Middleware
 * 
 * Ensures that the same request (identified by Idempotency-Key)
 * always produces the same result, even if sent multiple times.
 * 
 * Caches successful responses for 24 hours.
 */

const redis = new Redis({
  host: process.env.REDIS_HOST || 'localhost',
  port: parseInt(process.env.REDIS_PORT || '6379'),
  password: process.env.REDIS_PASSWORD,
  retryStrategy: (times) => Math.min(times * 50, 2000),
});

const IDEMPOTENCY_TTL = 24 * 3600; // 24 hours
const IDEMPOTENCY_PREFIX = 'idem:';

interface IdempotencyRequest extends Request {
  idempotencyKey?: string;
  originalJson?: Function;
}

export async function idempotencyMiddleware(
  req: IdempotencyRequest,
  res: Response,
  next: NextFunction
): Promise<void> {
  // Only apply to mutating methods
  if (['GET', 'HEAD', 'OPTIONS'].includes(req.method)) {
    return next();
  }

  const idempotencyKey = req.headers['x-idempotency-key'] as string;

  if (!idempotencyKey) {
    res.status(400).json({
      error: 'X-Idempotency-Key header is required for mutating requests',
    });
    return;
  }

  // Validate key format (prevent injection)
  if (!/^[a-zA-Z0-9_-]{8,128}$/.test(idempotencyKey)) {
    res.status(400).json({
      error: 'Idempotency-Key must be 8-128 alphanumeric characters',
    });
    return;
  }

  const cacheKey = `${IDEMPOTENCY_PREFIX}${idempotencyKey}`;

  try {
    // Check if this request was already processed
    const cachedResponse = await redis.get(cacheKey);

    if (cachedResponse) {
      const parsed = JSON.parse(cachedResponse);
      console.log(`Idempotency hit for key: ${idempotencyKey}`);
      
      // Return the exact same response
      res.status(parsed.statusCode).json(parsed.body);
      return;
    }

    // Mark as in-progress to prevent concurrent processing
    const lockAcquired = await redis.set(
      `${cacheKey}:lock`,
      'processing',
      'EX',
      30, // 30 second lock
      'NX'
    );

    if (!lockAcquired) {
      // Another request is processing this key
      // Wait and retry
      await new Promise(resolve => setTimeout(resolve, 100));
      
      const retryResponse = await redis.get(cacheKey);
      if (retryResponse) {
        const parsed = JSON.parse(retryResponse);
        res.status(parsed.statusCode).json(parsed.body);
        return;
      }
      
      res.status(409).json({
        error: 'Request is being processed. Please retry.',
      });
      return;
    }

    req.idempotencyKey = idempotencyKey;

    // Intercept the response to cache it
    const originalJson = res.json.bind(res);
    res.json = function (body: any) {
      // Cache successful responses
      if (res.statusCode >= 200 && res.statusCode < 500) {
        const cacheData = {
          statusCode: res.statusCode,
          body: body,
          cachedAt: new Date().toISOString(),
        };

        redis.setex(cacheKey, IDEMPOTENCY_TTL, JSON.stringify(cacheData))
          .catch(err => console.error('Failed to cache idempotent response:', err));

        // Release lock
        redis.del(`${cacheKey}:lock`).catch(() => {});
      } else {
        // Don't cache error responses, release lock
        redis.del(`${cacheKey}:lock`).catch(() => {});
      }

      return originalJson(body);
    };

    next();
  } catch (error) {
    // Release lock on error
    redis.del(`${cacheKey}:lock`).catch(() => {});
    console.error('Idempotency middleware error:', error);
    next(error);
  }
}

/**
 * Cleanup expired idempotency keys (runs periodically)
 */
export async function cleanupIdempotencyKeys(): Promise<void> {
  // Redis handles TTL automatically with EXPIRE
  // This is a safety measure for any keys without TTL
  const keys = await redis.keys(`${IDEMPOTENCY_PREFIX}*`);
  
  for (const key of keys) {
    if (!key.endsWith(':lock')) {
      const ttl = await redis.ttl(key);
      if (ttl === -1) { // No TTL set
        await redis.expire(key, IDEMPOTENCY_TTL);
      }
    }
  }
}
```

```bash
#!/bin/bash
# scripts/bootstrap.sh
# Single-command bootstrap for Nexa Pay Institutional Network

set -euo pipefail

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

echo -e "${BLUE}"
cat << "EOF"
╔══════════════════════════════════════════════════════════╗
║                                                          ║
║   NEXA PAY — Institutional Financial Network             ║
║   Powered by NexaToken                                   ║
║                                                          ║
║   Bootstrap Script v1.0                                  ║
║                                                          ║
╚══════════════════════════════════════════════════════════╝
EOF
echo -e "${NC}"

# Check prerequisites
check_prerequisites() {
    echo -e "${YELLOW}[1/8] Checking prerequisites...${NC}"
    
    MISSING=()
    
    command -v docker >/dev/null 2>&1 || MISSING+=("docker")
    command -v kubectl >/dev/null 2>&1 || MISSING+=("kubectl")
    command -v terraform >/dev/null 2>&1 || MISSING+=("terraform")
    command -v rustup >/dev/null 2>&1 || MISSING+=("rust")
    command -v go >/dev/null 2>&1 || MISSING+=("go")
    command -v node >/dev/null 2>&1 || MISSING+=("node")
    command -v cargo >/dev/null 2>&1 || MISSING+=("cargo")
    
    if [ ${#MISSING[@]} -gt 0 ]; then
        echo -e "${RED}Missing prerequisites: ${MISSING[*]}${NC}"
        exit 1
    fi
    
    echo -e "${GREEN}✓ All prerequisites installed${NC}"
}

# Setup environment
setup_env() {
    echo -e "${YELLOW}[2/8] Setting up environment...${NC}"
    
    if [ ! -f .env ]; then
        echo -e "${RED}.env file not found. Creating from template...${NC}"
        cat > .env << 'ENVEOF'
# Nexa Pay Environment Configuration
NODE_ENV=production
DATABASE_URL=postgresql://nexapay:changeme@localhost:5432/nexapay
REDIS_URL=redis://localhost:6379
API_SIGNING_SECRET=changeme_minimum_32_characters_here
MASTER_ENCRYPTION_KEY=changeme_64_hex_characters_here_0123456789abcdef
JWT_SECRET=changeme_minimum_32_characters_here
NEXA_GOVERNANCE_KEY=changeme_ed25519_private_key_here
ENVEOF
        echo -e "${RED}Please edit .env with your configuration before continuing${NC}"
        exit 1
    fi
    
    set -a
    source .env
    set +a
    
    echo -e "${GREEN}✓ Environment configured${NC}"
}

# Build all services
build_services() {
    echo -e "${YELLOW}[3/8] Building services...${NC}"
    
    # Build Rust services
    echo "  Building ledger-engine..."
    cd services/ledger-engine
    cargo build --release
    cd ../..
    
    echo "  Building token-engine..."
    cd services/token-engine
    cargo build --release
    cd ../..
    
    echo "  Building crypto-lib..."
    cd libs/crypto-lib
    cargo build --release
    cd ../..
    
    # Build Go services
    echo "  Building payment-engine..."
    cd services/payment-engine
    go build -o bin/payment-engine ./cmd/main.go
    cd ../..
    
    # Build Node.js services
    echo "  Building api-gateway..."
    cd services/api-gateway
    npm ci --production
    npm run build
    cd ../..
    
    echo "  Building webhook-service..."
    cd services/webhook-service
    npm ci --production
    npm run build
    cd ../..
    
    echo -e "${GREEN}✓ All services built${NC}"
}

# Build Docker images
build_images() {
    echo -e "${YELLOW}[4/8] Building Docker images...${NC}"
    
    docker build -t nexapay-ledger-engine:latest -f services/ledger-engine/Dockerfile .
    docker build -t nexapay-token-engine:latest -f services/token-engine/Dockerfile .
    docker build -t nexapay-payment-engine:latest -f services/payment-engine/Dockerfile .
    docker build -t nexapay-api-gateway:latest -f services/api-gateway/Dockerfile .
    docker build -t nexapay-webhook:latest -f services/webhook-service/Dockerfile .
    
    echo -e "${GREEN}✓ Docker images built${NC}"
}

# Initialize database
init_database() {
    echo -e "${YELLOW}[5/8] Initializing database...${NC}"
    
    # Run Prisma migrations
    cd services/api-gateway
    npx prisma migrate deploy
    npx prisma generate
    cd ../..
    
    # Seed initial data
    echo "  Seeding governance wallet..."
    # In production, this would create the governance wallet
    # and mint the initial token supply
    
    echo -e "${GREEN}✓ Database initialized${NC}"
}

# Start local cluster
start_cluster() {
    echo -e "${YELLOW}[6/8] Starting local cluster...${NC}"
    
    # Start infrastructure
    docker-compose -f infrastructure/docker/docker-compose.yml up -d
    
    # Wait for services
    echo "  Waiting for services to be ready..."
    sleep 10
    
    # Health checks
    for service in postgres redis; do
        if docker-compose -f infrastructure/docker/docker-compose.yml ps $service | grep -q "Up"; then
            echo -e "  ${GREEN}✓ $service is running${NC}"
        else
            echo -e "  ${RED}✗ $service failed to start${NC}"
            exit 1
        fi
    done
    
    echo -e "${GREEN}✓ Local cluster running${NC}"
}

# Run tests
run_tests() {
    echo -e "${YELLOW}[7/8] Running test suite...${NC}"
    
    # Rust tests
    echo "  Running ledger-engine tests..."
    cd services/ledger-engine
    cargo test --release
    cd ../..
    
    echo "  Running token-engine tests..."
    cd services/token-engine
    cargo test --release
    cd ../..
    
    # Go tests
    echo "  Running payment-engine tests..."
    cd services/payment-engine
    go test ./...
    cd ../..
    
    echo -e "${GREEN}✓ All tests passed${NC}"
}

# Verify deployment
verify_deployment() {
    echo -e "${YELLOW}[8/8] Verifying deployment...${NC}"
    
    # Check ledger engine
    echo "  Checking ledger engine..."
    if grpcurl -plaintext localhost:50051 ledger.LedgerService/GetStatus > /dev/null 2>&1; then
        echo -e "  ${GREEN}✓ Ledger engine healthy${NC}"
    else
        echo -e "  ${RED}✗ Ledger engine not responding${NC}"
    fi
    
    # Check token engine
    echo "  Checking token engine..."
    if grpcurl -plaintext localhost:50061 token.TokenService/GetSupply > /dev/null 2>&1; then
        echo -e "  ${GREEN}✓ Token engine healthy${NC}"
    else
        echo -e "  ${RED}✗ Token engine not responding${NC}"
    fi
    
    # Check API gateway
    echo "  Checking API gateway..."
    if curl -sf http://localhost:3001/health > /dev/null 2>&1; then
        echo -e "  ${GREEN}✓ API gateway healthy${NC}"
    else
        echo -e "  ${RED}✗ API gateway not responding${NC}"
    fi
}

# Main bootstrap flow
main() {
    check_prerequisites
    setup_env
    build_services
    build_images
    init_database
    start_cluster
    run_tests
    verify_deployment
    
    echo ""
    echo -e "${GREEN}══════════════════════════════════════════════════════════${NC}"
    echo -e "${GREEN}  Nexa Pay Institutional Network is LIVE${NC}"
    echo -e "${GREEN}══════════════════════════════════════════════════════════${NC}"
    echo ""
    echo -e "  Ledger Engine:    ${BLUE}grpc://localhost:50051${NC}"
    echo -e "  Token Engine:     ${BLUE}grpc://localhost:50061${NC}"
    echo -e "  Payment Engine:   ${BLUE}grpc://localhost:50071${NC}"
    echo -e "  API Gateway:      ${BLUE}https://localhost:3001${NC}"
    echo -e "  Webhook Service:  ${BLUE}https://localhost:3002${NC}"
    echo ""
    echo -e "  Next steps:"
    echo -e "  1. Configure your governance wallet"
    echo -e "  2. Mint initial NexaToken supply"
    echo -e "  3. Create merchant accounts"
    echo -e "  4. Start processing payments"
    echo ""
}

main "$@"
```
