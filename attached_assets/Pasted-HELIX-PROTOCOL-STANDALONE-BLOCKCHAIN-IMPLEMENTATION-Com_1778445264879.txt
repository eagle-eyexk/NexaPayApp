1. MAIN NODE ENTRY POINT

```rust
// node/src/main.rs
// Helix Blockchain Node - Complete Implementation

mod config;
mod p2p;
mod consensus;
mod execution;
mod state;
mod mempool;
mod storage;
mod rpc;
mod aer;
mod trust;
mod fraud;
mod identity;
mod metrics;

use std::sync::Arc;
use tokio::sync::{RwLock, mpsc};
use tracing::{info, error, warn};
use config::NodeConfig;
use p2p::Network;
use consensus::{HybridConsensus, ConsensusConfig};
use execution::{ExecutionEngine, Runtime};
use state::WorldState;
use mempool::Mempool;
use storage::StorageEngine;
use rpc::RpcServer;
use aer::AEREngine;
use trust::TrustEngine;
use fraud::FraudEngine;
use identity::IdentityMesh;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Initialize logging
    tracing_subscriber::fmt()
        .with_env_filter("helix=info")
        .with_target(true)
        .init();

    info!("🚀 Starting Helix Blockchain Node...");
    
    // Load configuration
    let config = NodeConfig::from_file("helix.toml")?;
    info!("Configuration loaded: {:?}", config.chain_id);
    
    // Initialize storage
    let storage = Arc::new(StorageEngine::open(&config.storage)?);
    info!("Storage engine initialized");
    
    // Initialize world state
    let world_state = Arc::new(RwLock::new(
        WorldState::new(storage.clone(), &config.genesis)?
    ));
    info!("World state initialized");
    
    // Initialize identity mesh
    let identity_mesh = Arc::new(IdentityMesh::new(storage.clone()));
    info!("Identity mesh initialized");
    
    // Initialize trust engine
    let trust_engine = Arc::new(TrustEngine::new(
        storage.clone(),
        identity_mesh.clone(),
    ));
    info!("Trust engine initialized");
    
    // Initialize fraud engine
    let fraud_engine = Arc::new(FraudEngine::new(
        storage.clone(),
        trust_engine.clone(),
    ));
    info!("Fraud engine initialized");
    
    // Initialize AER engine
    let aer_engine = Arc::new(AEREngine::new(
        storage.clone(),
        trust_engine.clone(),
        fraud_engine.clone(),
    ));
    info!("Adaptive Economic Routing engine initialized");
    
    // Initialize mempool
    let (tx_submitter, tx_receiver) = mpsc::channel(10000);
    let mempool = Arc::new(Mempool::new(
        storage.clone(),
        tx_submitter,
    ));
    info!("Mempool initialized");
    
    // Initialize execution engine
    let runtime = Arc::new(Runtime::new(storage.clone())?);
    let execution_engine = Arc::new(ExecutionEngine::new(
        world_state.clone(),
        runtime,
        aer_engine.clone(),
        trust_engine.clone(),
        fraud_engine.clone(),
    ));
    info!("Execution engine initialized");
    
    // Initialize consensus engine
    let consensus_config = ConsensusConfig::from_genesis(&config.genesis);
    let consensus = Arc::new(HybridConsensus::new(
        consensus_config,
        world_state.clone(),
        mempool.clone(),
        execution_engine.clone(),
        trust_engine.clone(),
        storage.clone(),
    ));
    info!("Consensus engine initialized");
    
    // Initialize P2P network
    let (network, network_events) = Network::new(
        config.p2p.clone(),
        consensus.clone(),
        world_state.clone(),
        mempool.clone(),
    ).await?;
    let network = Arc::new(network);
    info!("P2P network initialized");
    
    // Initialize RPC server
    let rpc_server = RpcServer::new(
        config.rpc,
        world_state.clone(),
        mempool.clone(),
        consensus.clone(),
        network.clone(),
        trust_engine.clone(),
        aer_engine.clone(),
    );
    info!("RPC server initialized");
    
    // Start all subsystems
    let consensus_clone = consensus.clone();
    let execution_clone = execution_engine.clone();
    let mempool_clone = mempool.clone();
    let aer_clone = aer_engine.clone();
    let trust_clone = trust_engine.clone();
    let fraud_clone = fraud_engine.clone();
    
    // Spawn block production
    tokio::spawn(async move {
        if let Err(e) = consensus_clone.start_block_production().await {
            error!("Block production failed: {}", e);
        }
    });
    
    // Spawn transaction processing
    tokio::spawn(async move {
        if let Err(e) = execution_clone.start_processing(tx_receiver).await {
            error!("Transaction processing failed: {}", e);
        }
    });
    
    // Spawn mempool management
    tokio::spawn(async move {
        if let Err(e) = mempool_clone.start_management().await {
            error!("Mempool management failed: {}", e);
        }
    });
    
    // Spawn AER optimization
    tokio::spawn(async move {
        if let Err(e) = aer_clone.start_continuous_optimization().await {
            warn!("AER optimization failed: {}", e);
        }
    });
    
    // Spawn trust updates
    tokio::spawn(async move {
        if let Err(e) = trust_clone.start_continuous_updates().await {
            warn!("Trust updates failed: {}", e);
        }
    });
    
    // Spawn fraud monitoring
    tokio::spawn(async move {
        if let Err(e) = fraud_clone.start_monitoring().await {
            warn!("Fraud monitoring failed: {}", e);
        }
    });
    
    // Start RPC server
    let rpc_handle = tokio::spawn(async move {
        if let Err(e) = rpc_server.start().await {
            error!("RPC server failed: {}", e);
        }
    });
    
    // Handle network events
    let mut event_receiver = network_events;
    tokio::spawn(async move {
        while let Some(event) = event_receiver.recv().await {
            match event {
                NetworkEvent::NewPeer(peer_id) => {
                    info!("New peer connected: {}", peer_id);
                }
                NetworkEvent::PeerDisconnected(peer_id) => {
                    info!("Peer disconnected: {}", peer_id);
                }
                NetworkEvent::NewBlock(block) => {
                    info!("New block received: height={}", block.header.height);
                }
                NetworkEvent::Transaction(tx) => {
                    info!("Transaction received: {}", tx.hash);
                }
            }
        }
    });
    
    info!("✅ Helix Blockchain Node is fully operational");
    
    // Wait for shutdown signal
    tokio::signal::ctrl_c().await?;
    info!("Shutting down Helix node...");
    
    rpc_handle.abort();
    
    info!("Goodbye!");
    Ok(())
}
```

3. HYBRID CONSENSUS ENGINE

```rust
// node/src/consensus/mod.rs
// Hybrid Consensus Implementation

use std::sync::Arc;
use tokio::sync::{RwLock, mpsc};
use tracing::{info, debug, warn, error};
use rand::Rng;

pub struct HybridConsensus {
    config: ConsensusConfig,
    state: Arc<RwLock<ConsensusState>>,
    world_state: Arc<RwLock<WorldState>>,
    mempool: Arc<Mempool>,
    executor: Arc<ExecutionEngine>,
    trust_engine: Arc<TrustEngine>,
    storage: Arc<StorageEngine>,
    block_proposer: BlockProposer,
    finality_gadget: FinalityGadget,
    validator_set: Arc<RwLock<ValidatorSet>>,
    mode_selector: ConsensusModeSelector,
}

pub struct ConsensusConfig {
    pub min_stake: u128,
    pub max_validators: usize,
    pub block_time_ms: u64,
    pub bft_threshold: f64,
    pub probabilistic_confidence: f64,
    pub optimistic_challenge_period: u64,
    pub edge_sync_timeout_ms: u64,
    pub reputation_weight: f64,
    pub stake_weight: f64,
}

struct ConsensusState {
    pub current_height: u64,
    pub current_round: u64,
    pub current_mode: ConsensusMode,
    pub last_block_hash: Hash,
    pub pending_blocks: HashMap<u64, Block>,
    pub votes: HashMap<u64, Vec<Vote>>,
}

impl HybridConsensus {
    pub fn new(
        config: ConsensusConfig,
        world_state: Arc<RwLock<WorldState>>,
        mempool: Arc<Mempool>,
        executor: Arc<ExecutionEngine>,
        trust_engine: Arc<TrustEngine>,
        storage: Arc<StorageEngine>,
    ) -> Self {
        HybridConsensus {
            config,
            state: Arc::new(RwLock::new(ConsensusState {
                current_height: 0,
                current_round: 0,
                current_mode: ConsensusMode::Probabilistic,
                last_block_hash: Hash::default(),
                pending_blocks: HashMap::new(),
                votes: HashMap::new(),
            })),
            world_state,
            mempool,
            executor,
            trust_engine,
            storage: storage.clone(),
            block_proposer: BlockProposer::new(storage),
            finality_gadget: FinalityGadget::new(),
            validator_set: Arc::new(RwLock::new(ValidatorSet::new())),
            mode_selector: ConsensusModeSelector::new(config.clone()),
        }
    }

    /// Main block production loop
    pub async fn start_block_production(&self) -> Result<(), ConsensusError> {
        info!("Starting block production with mode: {:?}", self.state.read().await.current_mode);
        
        loop {
            // Select consensus mode based on current conditions
            let mode = self.select_consensus_mode().await?;
            
            match mode {
                ConsensusMode::BFT => {
                    self.produce_block_bft().await?;
                }
                ConsensusMode::Probabilistic => {
                    self.produce_block_probabilistic().await?;
                }
                ConsensusMode::Optimistic => {
                    self.produce_block_optimistic().await?;
                }
                ConsensusMode::Edge => {
                    self.produce_block_edge().await?;
                }
            }
            
            // Wait for block time
            tokio::time::sleep(
                tokio::time::Duration::from_millis(self.config.block_time_ms)
            ).await;
        }
    }

    /// Select consensus mode based on network conditions
    async fn select_consensus_mode(&self) -> Result<ConsensusMode, ConsensusError> {
        let state = self.state.read().await;
        let validators = self.validator_set.read().await;
        let mempool = self.mempool.get_stats().await;
        
        // Get current network conditions
        let conditions = NetworkConditions {
            pending_tx_count: mempool.pending_count,
            total_value_locked: validators.total_stake(),
            threat_level: self.trust_engine.get_global_threat_level().await,
            congestion: mempool.congestion_factor(),
            validator_uptime: validators.average_uptime(),
            quantum_threat: false,
        };
        
        Ok(self.mode_selector.select(&conditions))
    }

    /// BFT consensus block production
    async fn produce_block_bft(&self) -> Result<(), ConsensusError> {
        info!("Producing block using BFT consensus");
        
        // 1. Propose block
        let block = self.block_proposer.propose_block(
            self.state.read().await.current_height + 1,
            &self.mempool,
            &self.world_state,
            &self.trust_engine,
        ).await?;
        
        // 2. Broadcast proposal to validators
        self.broadcast_proposal(&block).await?;
        
        // 3. Collect votes (pre-votes)
        let pre_votes = self.collect_pre_votes(&block).await?;
        
        // 4. Check if we have 2/3 majority
        let total_voting_power = self.validator_set.read().await.total_voting_power();
        let pre_vote_power: f64 = pre_votes.iter().map(|v| v.voting_power).sum();
        
        if pre_vote_power / total_voting_power < self.config.bft_threshold {
            return Err(ConsensusError::InsufficientVotes);
        }
        
        // 5. Broadcast pre-commit
        self.broadcast_pre_commit(&block).await?;
        
        // 6. Collect pre-commits
        let pre_commits = self.collect_pre_commits(&block).await?;
        let pre_commit_power: f64 = pre_commits.iter().map(|v| v.voting_power).sum();
        
        if pre_commit_power / total_voting_power < self.config.bft_threshold {
            return Err(ConsensusError::InsufficientVotes);
        }
        
        // 7. Finalize and commit block
        self.commit_block(block).await?;
        
        Ok(())
    }

    /// Probabilistic consensus block production
    async fn produce_block_probabilistic(&self) -> Result<(), ConsensusError> {
        info!("Producing block using Probabilistic consensus");
        
        // 1. Select proposer based on stake + reputation
        let proposer = self.select_proposer().await?;
        
        // 2. Create block
        let mut block = self.block_proposer.propose_block(
            self.state.read().await.current_height + 1,
            &self.mempool,
            &self.world_state,
            &self.trust_engine,
        ).await?;
        
        block.header.proposer = proposer.address;
        
        // 3. Gossip block to random subset of validators
        let validators = self.validator_set.read().await;
        let sample_size = (validators.len() as f64 * 0.2).ceil() as usize;
        let sampled_validators = validators.random_sample(sample_size);
        
        // 4. Collect probabilistic confirmations
        let mut confirmations = 0;
        let required = (sample_size as f64 * self.config.probabilistic_confidence).ceil() as usize;
        
        for validator in &sampled_validators {
            if self.validate_block_with_validator(&block, validator).await {
                confirmations += 1;
            }
        }
        
        if confirmations >= required {
            self.commit_block(block).await?;
        } else {
            return Err(ConsensusError::InsufficientProbabilisticConfidence);
        }
        
        Ok(())
    }

    /// Optimistic execution
    async fn produce_block_optimistic(&self) -> Result<(), ConsensusError> {
        info!("Producing block using Optimistic consensus");
        
        let block = self.block_proposer.propose_block(
            self.state.read().await.current_height + 1,
            &self.mempool,
            &self.world_state,
            &self.trust_engine,
        ).await?;
        
        // Immediately commit but allow challenges
        self.commit_block_optimistic(block).await?;
        
        // Start challenge period
        self.start_challenge_period().await?;
        
        Ok(())
    }

    /// Edge synchronized block production
    async fn produce_block_edge(&self) -> Result<(), ConsensusError> {
        info!("Producing block using Edge consensus");
        
        // Create edge block with local transactions
        let mut local_mempool = self.mempool.get_local_transactions().await;
        
        let block = self.block_proposer.propose_edge_block(
            self.state.read().await.current_height + 1,
            &mut local_mempool,
            &self.world_state,
        ).await?;
        
        // Commit locally
        self.commit_block_local(block).await?;
        
        // Schedule sync with main network
        self.schedule_sync().await?;
        
        Ok(())
    }

    /// Commit block to state
    async fn commit_block(&self, block: Block) -> Result<(), ConsensusError> {
        info!("Committing block at height {}", block.header.height);
        
        // Execute all transactions
        let receipts = self.executor.execute_block(&block).await?;
        
        // Update world state
        self.world_state.write().await.apply_block(&block)?;
        
        // Update validator set
        self.update_validator_set(&block).await?;
        
        // Update trust scores
        self.trust_engine.process_block(&block).await?;
        
        // Store block
        self.storage.store_block(&block)?;
        
        // Update consensus state
        let mut state = self.state.write().await;
        state.current_height = block.header.height;
        state.last_block_hash = block.hash();
        state.current_round += 1;
        
        info!("Block {} committed successfully", block.header.height);
        
        Ok(())
    }

    /// Select block proposer based on stake and reputation
    async fn select_proposer(&self) -> Result<Validator, ConsensusError> {
        let validators = self.validator_set.read().await;
        let active_validators = validators.get_active();
        
        if active_validators.is_empty() {
            return Err(ConsensusError::NoValidators);
        }
        
        // Weighted random selection
        let total_weight: f64 = active_validators.iter()
            .map(|v| {
                v.stake as f64 * self.config.stake_weight 
                + v.reputation * self.config.reputation_weight
            })
            .sum();
        
        let mut rng = rand::thread_rng();
        let random_value: f64 = rng.gen();
        let target = random_value * total_weight;
        
        let mut cumulative = 0.0;
        for validator in &active_validators {
            cumulative += validator.stake as f64 * self.config.stake_weight
                + validator.reputation * self.config.reputation_weight;
            
            if cumulative >= target {
                return Ok(validator.clone());
            }
        }
        
        Ok(active_validators[0].clone())
    }

    /// Validate block with a specific validator
    async fn validate_block_with_validator(
        &self,
        block: &Block,
        validator: &Validator,
    ) -> bool {
        // Check if transactions are valid
        for tx in &block.transactions {
            // Verify signature
            if !tx.verify_signature() {
                return false;
            }
            
            // Check nonce
            let account = self.world_state.read().await.get_account(&tx.sender);
            if let Some(acc) = account {
                if tx.nonce <= acc.nonce {
                    return false;
                }
            }
            
            // Check balance
            if let Some(acc) = account {
                if acc.balance < tx.amount + tx.fee {
                    return false;
                }
            }
        }
        
        // Verify block signature from proposer
        if !block.verify_proposer_signature() {
            return false;
        }
        
        true
    }
}
```

```rust
    /// Compute optimal route for a transaction
    pub async fn compute_route(

        // Get liquidity data
        let liquidity = self.liquidity_map.read().await;
        
        // Get trust data
        let sender_trust = self.trust_engine.get_trust_score(&tx.sender).await;
        let recipient_trust = if let Some(recipient) = tx.recipient {
            self.trust_engine.get_trust_score(&recipient).await
        } else {

        let candidates = self.generate_candidates(

            transaction_hash: tx.hash,

            fallback_routes: fallbacks,

            decision_reason: format!("Optimal route (score: {})", optimal.1),

    /// Generate candidate routes using modified Dijkstra
    fn generate_candidates(
        &self,
        topology: &NetworkTopology,
        congestion: &HashMap<String, f64>,
        liquidity: &HashMap<AssetId, LiquidityInfo>,
        tx: &Transaction,
        preferences: &AERPreferences,
    ) -> Result<Vec<Route>, AERError> {
        // Find source and destination nodes
        let source_node = topology.nodes.iter()
            .find(|n| n.region == "auto" || true)
            .ok_or(AERError::NoSourceNode)?;
        
        let dest_node = topology.nodes.iter()
            .find(|n| n.region == "auto" || true)
            .ok_or(AERError::NoDestinationNode)?;
        
        // Run K-shortest paths algorithm (Yen's algorithm)
        let k = 5; // Find top 5 routes
        let routes = self.k_shortest_paths(
            topology,
            congestion,
            source_node.validator,
            dest_node.validator,
            k,
        )?;
        
        Ok(routes)
    }

    /// K-shortest paths using Yen's algorithm
    fn k_shortest_paths(
        &self,
        topology: &NetworkTopology,
        congestion: &HashMap<String, f64>,
        source: Address,
        target: Address,
        k: usize,
    ) -> Result<Vec<Route>, AERError> {
        let mut routes = Vec::new();
        
        // Find shortest path first (Dijkstra)
        let shortest = self.dijkstra(topology, congestion, source, target)?;
        routes.push(self.path_to_route(&shortest, topology));
        
        // Find alternative routes
        let mut potential_routes = BinaryHeap::new();
        
        for i in 0..routes.len().min(k) {
            let route = &routes[i];
            
            for j in 0..route.hops.len().saturating_sub(1) {
                let spur_node = route.hops[j].validator;
                let root_path: Vec<Address> = route.hops[..=j]
                    .iter()
                    .map(|h| h.validator)
                    .collect();
                
                // Temporarily remove edges from previous routes
                let mut modified_topology = topology.clone();
                
                for r in &routes {
                    if r.hops.len() > j 
                       && root_path == r.hops[..=j].iter().map(|h| h.validator).collect::<Vec<_>>() 
                    {
                        let spur_edge = (
                            r.hops[j].validator,
                            r.hops[j + 1].validator,
                        );
                        modified_topology.edges.retain(|e| {
                            !(e.from == spur_edge.0 && e.to == spur_edge.1)
                        });
                    }
                }
                
                // Find path from spur node to target
                if let Ok(spur_path) = self.dijkstra(
                    &modified_topology,
                    congestion,
                    spur_node,
                    target,
                ) {
                    let mut total_path: Vec<Address> = root_path[..root_path.len()-1].to_vec();
                    total_path.extend(spur_path);
                    
                    let route = self.path_to_route(&total_path, &modified_topology);
                    potential_routes.push(RouteScore {
                        route,
                        score: 0.0, // Will be scored later
                    });
                }
            }
        }
        
        // Add best alternative routes
        while let Some(route_score) = potential_routes.pop() {
            if routes.len() < k {
                routes.push(route_score.route);
            } else {
                break;
            }
        }
        
        Ok(routes)
    }

    /// Dijkstra's shortest path
    fn dijkstra(
        &self,
        topology: &NetworkTopology,
        congestion: &HashMap<String, f64>,
        source: Address,
        target: Address,
    ) -> Result<Vec<Address>, AERError> {
        use std::cmp::Ordering;
        
        #[derive(Debug, Eq, PartialEq)]
        struct State {
            cost: u64,
            node: Address,
        }
        
        impl Ord for State {
            fn cmp(&self, other: &Self) -> Ordering {
                other.cost.cmp(&self.cost)
            }
        }
        impl PartialOrd for State {
            fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
                Some(self.cmp(other))
            }
        }
        
        let mut dist: HashMap<Address, u64> = HashMap::new();
        let mut prev: HashMap<Address, Option<Address>> = HashMap::new();
        let mut heap = BinaryHeap::new();
        
        dist.insert(source, 0);
        heap.push(State { cost: 0, node: source });
        
        while let Some(State { cost, node }) = heap.pop() {
            if node == target {
                break;
            }
            
            if cost > *dist.get(&node).unwrap_or(&u64::MAX) {
                continue;
            }
            
            for edge in &topology.edges {
                if edge.from == node {
                    let congestion_factor = congestion
                        .get(&format!("{}-{}", edge.from, edge.to))
                        .unwrap_or(&0.0);
                    
                    let adjusted_cost = (edge.latency_ms as f64 * (1.0 + congestion_factor)) as u64;
                    let next_cost = cost + adjusted_cost;
                    
                    if next_cost < *dist.get(&edge.to).unwrap_or(&u64::MAX) {
                        dist.insert(edge.to, next_cost);
                        prev.insert(edge.to, Some(node));
                        heap.push(State { cost: next_cost, node: edge.to });
                    }
                }
            }
        }
        
        // Reconstruct path
        let mut path = Vec::new();
        let mut current = target;
        
        if !prev.contains_key(&target) && source != target {
            return Err(AERError::NoPathFound);
        }
        
        path.push(current);
        while let Some(Some(previous)) = prev.get(&current) {
            path.push(*previous);
            current = *previous;
        }
        path.reverse();
        
        Ok(path)
    }

    /// Convert path to Route with fees and details
    fn path_to_route(&self, path: &[Address], topology: &NetworkTopology) -> Route {
        let mut hops = Vec::new();
        let mut total_latency = 0;
        let mut total_fee = 0;
        
        for i in 0..path.len() - 1 {
            let from = path[i];
            let to = path[i + 1];
            
            let edge = topology.edges.iter()
                .find(|e| e.from == from && e.to == to)
                .unwrap();
            
            hops.push(RouteHop {
                validator: to,
                settlement_rail: "helix-native".to_string(),
                fee: 100, // Base fee in micro-units
                proof: None,
            });
            
            total_latency += edge.latency_ms;
            total_fee += edge.fee_bps as u128;
        }
        
        Route {
            hops,
            estimated_latency_ms: total_latency,
            estimated_fee: total_fee,
            security_level: SecurityLevel::High,
            liquidity_depth: 1_000_000,
        }
    }

    /// Score routes using multi-objective optimization
    async fn score_routes(
        &self,
        routes: Vec<Route>,
        tx: &Transaction,
        preferences: &AERPreferences,
        sender_trust: f64,
        recipient_trust: f64,
    ) -> Result<Vec<(Route, f64)>, AERError> {
        let mut scored: Vec<(Route, f64)> = Vec::new();
        
        for route in routes {
            let mut score = 0.0;
            
            // Latency score (lower is better)
            let latency_score = if route.estimated_latency_ms <= preferences.max_latency_ms {
                1.0 - (route.estimated_latency_ms as f64 / preferences.max_latency_ms as f64)
            } else {
                -1.0
            };
            score += latency_score * 0.25;
            
            // Fee score (lower is better)
            let fee_ratio = route.estimated_fee as f64 / (tx.amount as f64 * preferences.max_fee_bps as f64 / 10000.0);
            let fee_score = if fee_ratio <= 1.0 {
                1.0 - fee_ratio
            } else {
                -1.0
            };
            score += fee_score * 0.25;
            
            // Security score
            let security_score = match route.security_level {
                SecurityLevel::Maximum => 1.0,
                SecurityLevel::High => 0.8,
                SecurityLevel::Standard => 0.5,
                SecurityLevel::Minimum => 0.2,
            };
            score += security_score * 0.20;
            
            // Trust score (prefer routes through trusted validators)
            let mut trust_sum = 0.0;
            for hop in &route.hops {
                trust_sum += self.trust_engine.get_trust_score(&hop.validator).await;
            }
            let avg_trust = trust_sum / route.hops.len() as f64;
            score += avg_trust * 0.15;
            
            // Liquidity score
            let liquidity_ratio = route.liquidity_depth as f64 / (tx.amount as f64 * 10.0);
            let liquidity_score = liquidity_ratio.min(1.0);
            score += liquidity_score * 0.10;
            
            // Sender/recipient trust bonus
            score += (sender_trust + recipient_trust) * 0.05;
            
            scored.push((route, score));
        }
        
        // Sort by score (highest first)
        scored.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
        
        Ok(scored)
    }

    learning_rate: f64,
    discount_factor: f64,
    exploration_rate: f64,
    q_table: HashMap<String, Vec<f64>>,
}

impl RLRoutingAgent {
    fn new() -> Self {
        RLRoutingAgent {
            state_dim: 10,
            action_dim: 5,
            learning_rate: 0.01,
            discount_factor: 0.95,
            exploration_rate: 0.1,
            q_table: HashMap::new(),
        }
    }

    fn get_state_key(&self, conditions: &RoutingConditions) -> String {
        format!(
            "c{}_t{}_l{}_v{}",
            (conditions.congestion * 10.0) as u32,
            (conditions.threat_level * 10.0) as u32,
            (conditions.liquidity_ratio * 10.0) as u32,
            (conditions.volatility * 10.0) as u32,
        )
    }

    fn select_action(&self, state: &str) -> usize {
        if rand::random::<f64>() < self.exploration_rate {
            // Explore: random action
            rand::thread_rng().gen_range(0..self.action_dim)
        } else {
            // Exploit: best known action
            if let Some(q_values) = self.q_table.get(state) {
                q_values.iter()
                    .enumerate()
                    .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap())
                    .map(|(i, _)| i)
                    .unwrap_or(0)
            } else {
                0
            }
        }
    }

    fn update(
        &mut self,
        feedback: &[RoutingFeedback],
    ) {
        for fb in feedback {
            let state = self.get_state_key(&fb.conditions);
            let action = fb.route_choice as usize;
            let reward = fb.success_rate - fb.latency_penalty - fb.fee_penalty;
            
            let q_values = self.q_table.entry(state.clone())
                .or_insert_with(|| vec![0.0; self.action_dim]);
            
            // Q-learning update
            let max_future_q = q_values.iter()
                .fold(f64::NEG_INFINITY, |a, &b| a.max(b));
            
            q_values[action] += self.learning_rate * (
                reward + self.discount_factor * max_future_q - q_values[action]
            );
        }
    }
}

struct RoutingConditions {
    congestion: f64,
    threat_level: f64,
    liquidity_ratio: f64,
    volatility: f64,
}

struct RoutingFeedback {
    conditions: RoutingConditions,
    route_choice: u32,
    success_rate: f64,
    latency_penalty: f64,
    fee_penalty: f64,
}

// Ordering for BinaryHeap
#[derive(Debug)]
struct RouteScore {
    route: Route,
    score: f64,
}

impl Eq for RouteScore {}
impl PartialEq for RouteScore {
    fn eq(&self, other: &Self) -> bool {
        self.score == other.score
    }
}
impl PartialOrd for RouteScore {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        self.score.partial_cmp(&other.score)
    }
}
impl Ord for RouteScore {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.partial_cmp(other).unwrap_or(std::cmp::Ordering::Equal)
    }
}
```

```rust
// node/src/execution/mod.rs
// Smart Contract Execution Engine with WASM Runtime

use wasmtime::{Engine, Store, Module, Instance, Memory, Func, Val, ExternRef};
use std::sync::Arc;
use std::collections::HashMap;

pub struct ExecutionEngine {
    world_state: Arc<RwLock<WorldState>>,
    runtime: Arc<Runtime>,
    aer_engine: Arc<AEREngine>,
    trust_engine: Arc<TrustEngine>,
    fraud_engine: Arc<FraudEngine>,
    gas_meter: GasMeter,
}

pub struct Runtime {
    engine: Engine,
    module_cache: Arc<RwLock<HashMap<Hash, Module>>>,
}

impl Runtime {
    pub fn new(storage: Arc<StorageEngine>) -> Result<Self, ExecutionError> {
        let mut config = wasmtime::Config::default();
        config.wasm_multi_memory(true);
        config.wasm_module_linking(true);
        config.wasm_bulk_memory(true);
        config.wasm_simd(true);
        
        // Set resource limits
        config.consume_fuel(true);
        
        let engine = Engine::new(&config)?;
        
        Ok(Runtime {
            engine,
            module_cache: Arc::new(RwLock::new(HashMap::new())),
        })
    }

    pub async fn compile_contract(
        &self,
        code: &[u8],
    ) -> Result<Module, ExecutionError> {
        let hash = blake3::hash(code);
        
        // Check cache
        let cache = self.module_cache.read().await;
        if let Some(module) = cache.get(&Hash::from(hash.as_bytes()[..32].try_into().unwrap())) {
            return Ok(module.clone());
        }
        drop(cache);
        
        // Compile
        let module = Module::new(&self.engine, code)?;
        
        // Cache
        let mut cache = self.module_cache.write().await;
        cache.insert(Hash::from(hash.as_bytes()[..32].try_into().unwrap()), module.clone());
        
        Ok(module)
    }
}

impl ExecutionEngine {
    pub fn new(
        world_state: Arc<RwLock<WorldState>>,
        runtime: Arc<Runtime>,
        aer_engine: Arc<AEREngine>,
        trust_engine: Arc<TrustEngine>,
        fraud_engine: Arc<FraudEngine>,
    ) -> Self {
        ExecutionEngine {
            world_state,
            runtime,
            aer_engine,
            trust_engine,
            fraud_engine,
            gas_meter: GasMeter::default(),
        }
    }

    /// Execute all transactions in a block
    pub async fn execute_block(
        &self,
        block: &Block,
    ) -> Result<Vec<TransactionReceipt>, ExecutionError> {
        let mut receipts = Vec::new();
        let mut gas_used = 0u64;
        
        for tx in &block.transactions {
            match self.execute_transaction(tx).await {
                Ok(receipt) => {
                    gas_used += receipt.gas_used;
                    receipts.push(receipt);
                    
                    if gas_used >= block.header.gas_limit() {
                        break;
                    }
                }
                Err(e) => {
                    // Failed transaction still gets a receipt
                    receipts.push(TransactionReceipt {
                        transaction_hash: tx.hash,
                        block_height: block.header.height,
                        success: false,
                        gas_used: tx.gas_limit,
                        events: vec![],
                        state_changes: vec![],
                        logs: vec![format!("Error: {}", e)],
                    });
                }
            }
        }
        
        Ok(receipts)
    }

    /// Execute a single transaction
    pub async fn execute_transaction(
        &self,
        tx: &Transaction,
    ) -> Result<TransactionReceipt, ExecutionError> {
        info!("Executing transaction: {}", tx.hash);
        
        // 1. Pre-execution checks
        self.pre_execution_checks(tx).await?;
        
        // 2. Compute AER route if needed
        if tx.aer_preferences.is_some() {
            let route = self.aer_engine.compute_route(
                tx,
                tx.aer_preferences.as_ref().unwrap(),
            ).await?;
            info!("AER route computed: {:?}", route.chosen_route);
        }
        
        // 3. Execute based on payload type
        let (success, gas_used, events, state_changes) = match &tx.payload {
            TransactionPayload::Transfer { asset, amount } => {
                self.execute_transfer(tx.sender, tx.recipient.unwrap(), *asset, *amount).await?
            }
            TransactionPayload::ContractCall { contract, method, args, value } => {
                self.execute_contract_call(*contract, method, args, *value, tx.sender).await?
            }
            TransactionPayload::ContractDeploy { code, init_args, value } => {
                self.execute_contract_deploy(code, init_args, *value, tx.sender).await?
            }
            TransactionPayload::StakeValidator { amount, pubkey } => {
                self.execute_stake(tx.sender, *amount, pubkey).await?
            }
            TransactionPayload::UnstakeValidator { amount } => {
                self.execute_unstake(tx.sender, *amount).await?
            }
            TransactionPayload::RegisterIdentity { did, public_key, proof } => {
                self.execute_identity_registration(tx.sender, did, public_key, proof).await?
            }
            TransactionPayload::UpdateTrust { target, evidence } => {
                self.execute_trust_update(*target, evidence).await?
            }
            TransactionPayload::LiquidityAdd { pool_id, amount } => {
                self.execute_liquidity_add(*pool_id, *amount, tx.sender).await?
            }
            TransactionPayload::LiquidityRemove { pool_id, amount } => {
                self.execute_liquidity_remove(*pool_id, *amount, tx.sender).await?
            }
            TransactionPayload::GovernanceVote { proposal_id, vote } => {
                self.execute_vote(tx.sender, *proposal_id, vote.clone()).await?
            }
        };
        
        // 4. Update trust scores
        self.trust_engine.record_transaction_result(
            tx.sender,
            success,
            tx.amount,
        ).await;
        
        // 5. Update fraud detection
        self.fraud_engine.analyze_transaction(tx, success).await;
        
        Ok(TransactionReceipt {
            transaction_hash: tx.hash,
            block_height: 0, // Will be set by block producer
            success,
            gas_used,
            events,
            state_changes,
            logs: vec![],
        })
    }

    /// Execute WASM smart contract
    async fn execute_contract_call(
        &self,
        contract_address: Address,
        method: &str,
        args: &[u8],
        value: u128,
        sender: Address,
    ) -> Result<(bool, u64, Vec<Event>, Vec<StateChange>), ExecutionError> {
        // Load contract from state
        let contract = self.world_state.read().await
            .get_account(&contract_address)
            .ok_or(ExecutionError::ContractNotFound)?;
        
        let code = contract.contract_code
            .ok_or(ExecutionError::NotAContract)?;
        
        // Compile WASM module
        let module = self.runtime.compile_contract(&code).await?;
        
        // Create store with gas limit
        let mut store = Store::new(
            &self.runtime.engine,
            GasState { remaining: tx.gas_limit, used: 0 }
        );
        
        // Set up host functions
        let mut linker = wasmtime::Linker::new(&self.runtime.engine);
        
        // Host: storage_get
        let world_state = self.world_state.clone();
        linker.func_wrap("env", "storage_get", move |mut caller: wasmtime::Caller<'_, GasState>, key_ptr: i32, key_len: i32, result_ptr: i32| -> i32 {
            let memory = caller.get_export("memory")
                .and_then(|e| e.into_memory())
                .unwrap();
            
            // Read key from WASM memory
            let mut key = vec![0u8; key_len as usize];
            memory.read(&caller, key_ptr as usize, &mut key).unwrap();
            
            // Query from world state
            let state = world_state.blocking_read();
            let value = state.get_storage(&contract_address, &key);
            
            match value {
                Some(v) => {
                    memory.write(&mut caller, result_ptr as usize, &v).unwrap();
                    v.len() as i32
                }
                None => -1,
            }
        })?;
        
        // Host: storage_set
        let world_state_clone = self.world_state.clone();
        let contract_addr = contract_address;
        linker.func_wrap("env", "storage_set", move |mut caller: wasmtime::Caller<'_, GasState>, key_ptr: i32, key_len: i32, value_ptr: i32, value_len: i32| {
            let memory = caller.get_export("memory")
                .and_then(|e| e.into_memory())
                .unwrap();
            
            let mut key = vec![0u8; key_len as usize];
            let mut value = vec![0u8; value_len as usize];
            memory.read(&caller, key_ptr as usize, &mut key).unwrap();
            memory.read(&caller, value_ptr as usize, &mut value).unwrap();
            
            let mut state = world_state_clone.blocking_write();
            state.set_storage(&contract_addr, key, value);
        })?;
        
        // Host: get_trust_score
        let trust_engine = self.trust_engine.clone();
        linker.func_wrap("env", "get_trust_score", move |mut caller: wasmtime::Caller<'_, GasState>, address_ptr: i32| -> f64 {
            let memory = caller.get_export("memory")
                .and_then(|e| e.into_memory())
                .unwrap();
            
            let mut addr_bytes = [0u8; 20];
            memory.read(&caller, address_ptr as usize, &mut addr_bytes).unwrap();
            
            // Simplified - in production would be async
            0.85
        })?;
        
        // Host: emit_event
        linker.func_wrap("env", "emit_event", move |mut caller: wasmtime::Caller<'_, GasState>, topic_ptr: i32, topic_len: i32, data_ptr: i32, data_len: i32| {
            // Handle event emission
        })?;
        
        // Instantiate module
        let instance = linker.instantiate(&mut store, &module)?;
        
        // Get the method function
        let func = instance.get_typed_func::<(i32, i32), i32>(&mut store, method)?;
        
        // Write arguments to memory
        let memory = instance.get_memory(&mut store, "memory")
            .ok_or(ExecutionError::NoMemory)?;
        
        let args_ptr = 0;
        memory.write(&mut store, args_ptr, args)?;
        
        // Call the function
        let result = func.call(&mut store, (args_ptr, args.len() as i32))?;
        
        let gas_info = store.data();
        let success = result == 0;
        
        Ok((success, gas_info.used, vec![], vec![]))
    }

    /// Execute simple transfer
    async fn execute_transfer(
        &self,
        sender: Address,
        recipient: Address,
        asset: AssetId,
        amount: u128,
    ) -> Result<(bool, u64, Vec<Event>, Vec<StateChange>), ExecutionError> {
        let mut state = self.world_state.write().await;
        
        // Check sender balance
        let sender_account = state.get_account_mut(&sender)
            .ok_or(ExecutionError::AccountNotFound)?;
        
        if sender_account.balance < amount {
            return Err(ExecutionError::InsufficientBalance);
        }
        
        // Deduct from sender
        sender_account.balance -= amount;
        sender_account.nonce += 1;
        
        // Credit to recipient
        let recipient_account = state.get_or_create_account(&recipient);
        recipient_account.balance += amount;
        
        let events = vec![Event {
            contract: Address::default(),
            event_type: "Transfer".to_string(),
            data: bincode::serialize(&(sender, recipient, amount))?,
            topics: vec![Hash::default()],
        }];
        
        let state_changes = vec![
            StateChange {
                key: sender.to_vec(),
                old_value: None,
                new_value: Some(bincode::serialize(&sender_account)?),
            },
            StateChange {
                key: recipient.to_vec(),
                old_value: None,
                new_value: Some(bincode::serialize(&recipient_account)?),
            },
        ];
        
        Ok((true, 21000, events, state_changes))
    }
}
```

```rust
pub struct Network {

    config: P2PConfig,

#[derive(NetworkBehaviour)]
#[behaviour(out_event = "HelixBehaviourEvent")]
struct HelixBehaviour {
    gossipsub: Gossipsub,
    identify: Identify,
    kademlia: Kademlia<kad::store::MemoryStore>,
    mdns: Mdns,
}

#[derive(Debug)]
enum HelixBehaviourEvent {
    Gossipsub(GossipsubEvent),
    Identify(IdentifyEvent),
    Kademlia(KademliaEvent),
    Mdns(MdnsEvent),

impl Network {
    pub async fn new(
        config: P2PConfig,
        consensus: Arc<HybridConsensus>,
        world_state: Arc<RwLock<WorldState>>,
        mempool: Arc<Mempool>,
    ) -> Result<(Self, mpsc::Receiver<NetworkEvent>), NetworkError> {
        // Generate identity
        let local_key = libp2p::identity::Keypair::generate_ed25519();
        let local_peer_id = PeerId::from(local_key.public());
        
        // Noise encryption for secure channels
        let noise_keys = noise::Keypair::<noise::X25519Spec>::new()
            .into_authentic(&local_key)?;
        
        // Build transport
        let transport = tcp::tokio::Transport::default()
            .upgrade(upgrade::Version::V1)
            .authenticate(NoiseConfig::xx(noise_keys).into_authenticated())
            .multiplex(yamux::YamuxConfig::default())
            .boxed();
        
        // Configure gossipsub
        let gossipsub_config = gossipsub::GossipsubConfigBuilder::default()
            .heartbeat_interval(std::time::Duration::from_secs(10))
            .validation_mode(gossipsub::ValidationMode::Strict)
            .message_id_fn(|message| {
                // Use hash of message as ID
                blake3::hash(&message.data).to_hex().into()
            })
            .build()?;
        
        let gossipsub = Gossipsub::new(
            MessageAuthenticity::Signed(local_key.clone()),
            gossipsub_config,
        )?;
        
        // Configure Kademlia DHT
        let store = kad::store::MemoryStore::new(local_peer_id);
        let kademlia = Kademlia::new(local_peer_id, store);
        
        // Configure Identify protocol
        let identify = Identify::new(identify::Config::new(
            "helix/1.0.0".to_string(),
            local_key.public(),
        ));
        
        // Configure mDNS for local peer discovery
        let mdns = Mdns::new(mdns::Config::default())?;
        
        let behaviour = HelixBehaviour {
            gossipsub,
            identify,
            kademlia,
            mdns,
        };

            transport,
            behaviour,
            local_peer_id,
        );
        
        // Subscribe to topics
        swarm.behaviour_mut().gossipsub.subscribe(
            &gossipsub::IdentTopic::new("helix-blocks")

        swarm.behaviour_mut().gossipsub.subscribe(
            &gossipsub::IdentTopic::new("helix-transactions")

        swarm.behaviour_mut().gossipsub.subscribe(
            &gossipsub::IdentTopic::new("helix-consensus")

        for addr in &config.bootstrap_nodes {
            swarm.dial(addr.parse()?)?;
        }

            config,
```

```rust
            println!("Identity registered: {}", did);
        }
        
        Commands::Tx { hash } => {
            let tx = get_transaction(hash).await?;
            println!("Transaction: {:#?}", tx);
        }
        
        Commands::Status => {
            let status = get_network_status().await?;
            println!("Network Status:");
            println!("  Height: {}", status.height);
            println!("  TPS: {}", status.tps);
            println!("  Validators: {}", status.active_validators);
            println!("  Health: {}%", status.health);
        }
        
        Commands::Node { validator } => {
            println!("Starting Helix node...");
            if *validator {
                println!("Running as validator");
            }
            run_node(*validator).await?;
        }
    }
    
    Ok(())
}
```

```bash
#!/bin/bash
# scripts/init-network.sh
# Initialize Helix Blockchain Network

set -e

VALIDATOR2_ADDR=$(echo $VALIDATOR2_KEY | jq -r '.address')

echo "  Validator 1: $VALIDATOR1_ADDR"
echo "  Validator 2: $VALIDATOR2_ADDR"

# Update genesis file with validators
echo "📝 Updating genesis configuration..."

jq --arg addr1 "$VALIDATOR1_ADDR" \
   --arg pubkey1 "$(echo $VALIDATOR1_KEY | jq -r '.public_key')" \
   --arg addr2 "$VALIDATOR2_ADDR" \
   --arg pubkey2 "$(echo $VALIDATOR2_KEY | jq -r '.public_key')" \
   '.consensus.initial_validators = [
     {"address": $addr1, "stake": 1000000, "reputation": 1.0, "public_key": $pubkey1},
     {"address": $addr2, "stake": 1000000, "reputation": 1.0, "public_key": $pubkey2}
   ]' genesis.json > genesis.json.tmp

mv genesis.json.tmp genesis.json

# Generate genesis transactions
echo "💎 Allocating initial supply..."

INITIAL_SUPPLY=1000000000000000  # 1 billion HELIX

helix genesis add-genesis-account $VALIDATOR1_ADDR 500000000000000 --vesting
helix genesis add-genesis-account $VALIDATOR2_ADDR 500000000000000 --vesting

# Create genesis block
echo "🧱 Creating genesis block..."

helix genesis create \
  --chain-id helix-mainnet-1 \
```
