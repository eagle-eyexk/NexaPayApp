pub struct AdaptiveLedger {
    transactions: Arc<RwLock<HashMap<Uuid, HelixTransaction>>>,
    state_merkle_root: Arc<RwLock<String>>,
    event_log: Arc<RwLock<Vec<LedgerEvent>>>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum LedgerEvent {
    TransactionCreated(Uuid),
    RouteSelected(Uuid, AdaptiveRoute),
    TrustUpdated(Uuid, TrustVector),
    SettlementStarted(Uuid),
    SettlementCompleted(Uuid),
    FraudFlagged(Uuid, f64),
}

impl AdaptiveLedger {
    pub fn new() -> Self {
        AdaptiveLedger {
            transactions: Arc::new(RwLock::new(HashMap::new())),
            state_merkle_root: Arc::new(RwLock::new(String::new())),
            event_log: Arc::new(RwLock::new(Vec::new())),
        }
    }

    pub async fn record_transaction(&self, tx: HelixTransaction) -> Result<Uuid, String> {
        let tx_id = tx.id;
        let mut txs = self.transactions.write().await;
        txs.insert(tx_id, tx.clone());
        
        let mut events = self.event_log.write().await;
        events.push(LedgerEvent::TransactionCreated(tx_id));
        
        self.recompute_merkle_root().await;
        
        Ok(tx_id)
    }

    pub async fn update_trust(&self, tx_id: Uuid, trust: TrustVector) -> Result<(), String> {
        let mut txs = self.transactions.write().await;
        if let Some(tx) = txs.get_mut(&tx_id) {
            tx.trust_score = trust.clone();
            tx.status = TransactionStatus::Validating;
            
            let mut events = self.event_log.write().await;
            events.push(LedgerEvent::TrustUpdated(tx_id, trust));
        }
        Ok(())
    }

    async fn recompute_merkle_root(&self) {
        let txs = self.transactions.read().await;
        let mut hasher = Sha3_256::new();
        
        let mut tx_ids: Vec<&Uuid> = txs.keys().collect();
        tx_ids.sort();
        
        for id in tx_ids {
            if let Some(tx) = txs.get(id) {
                let tx_json = serde_json::to_string(tx).unwrap_or_default();
                hasher.update(tx_json.as_bytes());
            }
        }
        
        let result = hasher.finalize();
        let mut merkle_root = self.state_merkle_root.write().await;
        *merkle_root = hex::encode(result);
    }

    pub async fn get_state_root(&self) -> String {
        self.state_merkle_root.read().await.clone()
    }
}

    pub liquidity_weight: f64,
    pub reliability_weight: f64,
}

impl AEREngine {
    pub fn new() -> Self {
        AEREngine {
            network_graph: Arc::new(RwLock::new(NetworkGraph {
                nodes: Vec::new(),
                edges: Vec::new(),
                updated_at: Utc::now(),
            })),
            routing_weights: Arc::new(RwLock::new(RoutingWeights {
                latency_weight: 0.25,
                fee_weight: 0.25,
                security_weight: 0.20,
                liquidity_weight: 0.15,
                reliability_weight: 0.15,
            })),
            congestion_map: Arc::new(RwLock::new(HashMap::new())),
            liquidity_map: Arc::new(RwLock::new(HashMap::new())),
            ml_model: None,
        }
    }

    pub async fn compute_optimal_route(
        &self,
        source: &EntityDescriptor,
        destination: &EntityDescriptor,
        amount: &Amount,
    ) -> Result<AdaptiveRoute, String> {
        let graph = self.network_graph.read().await;
        let weights = self.routing_weights.read().await;
        let congestion = self.congestion_map.read().await;
        let liquidity = self.liquidity_map.read().await;

        // Generate candidate routes
        let candidates = self.generate_route_candidates(
            &graph,
            source,
            destination,
            amount,
        ).await?;

        // Score each candidate
        let mut scored_routes: Vec<(AdaptiveRoute, f64)> = Vec::new();
        
        for candidate in candidates {
            let score = self.score_route(
                &candidate,
                &weights,
                &congestion,
                &liquidity,
            ).await;
            scored_routes.push((candidate, score));
        }

        // Select optimal route
        scored_routes.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
        
        if let Some((best_route, _)) = scored_routes.first() {
            let mut route = best_route.clone();
            route.fallback_routes = scored_routes[1..]
                .iter()
                .take(3)
                .map(|(r, _)| AlternativeRoute {
                    hops: r.hops.clone(),
                    estimated_latency_ms: r.estimated_latency_ms,
                    estimated_fee: r.estimated_fee,
                })
                .collect();
            
            Ok(route)
        } else {
            Err("No viable route found".to_string())
        }
    }

    async fn generate_route_candidates(
        &self,
        graph: &NetworkGraph,
        source: &EntityDescriptor,
        destination: &EntityDescriptor,
        amount: &Amount,
    ) -> Result<Vec<AdaptiveRoute>, String> {
        let mut routes = Vec::new();
        
        // Find source and destination nodes
        let source_node = graph.nodes.iter()
            .find(|n| n.region == "source_region")
            .unwrap_or(&graph.nodes[0]);
        
        let dest_node = graph.nodes.iter()
            .find(|n| n.region == "dest_region")
            .unwrap_or(&graph.nodes[1]);

        // Dijkstra's algorithm for shortest path
        let path = self.dijkstra_shortest_path(graph, source_node.id, dest_node.id)?;
        
        let hops: Vec<RouteHop> = path.iter().map(|node_id| {
            let node = graph.nodes.iter().find(|n| &n.id == node_id).unwrap();
            RouteHop {
                node_id: node.id,
                node_type: node.node_type.clone(),
                settlement_rail: format!("rail_{}", node.id),
                fee: amount.value * 0.001,
                latency_ms: 100,
            }
        }).collect();

        let route = AdaptiveRoute {
            hops,
            chosen_by: RouteDecision::Optimal,
            estimated_latency_ms: 500,
            estimated_fee: amount.value * 0.01,
            liquidity_depth: 1000000.0,
            security_level: SecurityLevel::High,
            fallback_routes: Vec::new(),
            policies: Vec::new(),
        };

        routes.push(route);
        Ok(routes)
    }

    async fn score_route(
        &self,
        route: &AdaptiveRoute,
        weights: &RoutingWeights,
        congestion: &HashMap<String, f64>,
        liquidity: &HashMap<String, f64>,
    ) -> f64 {
        let latency_score = route.estimated_latency_ms as f64 / 1000.0;
        let fee_score = route.estimated_fee;
        let security_score = match route.security_level {
            SecurityLevel::High => 0.0,
            SecurityLevel::Medium => 0.5,
            SecurityLevel::Low => 1.0,
        };
        let liquidity_score = 1.0 / (route.liquidity_depth + 1.0);
        let reliability_score = 0.5; // Default

        weights.latency_weight * latency_score
            + weights.fee_weight * fee_score
            + weights.security_weight * security_score
            + weights.liquidity_weight * liquidity_score
            + weights.reliability_weight * reliability_score
    }

    fn dijkstra_shortest_path(
        &self,
        graph: &NetworkGraph,
        source: Uuid,
        target: Uuid,
    ) -> Result<Vec<Uuid>, String> {
        use std::collections::BinaryHeap;
        use std::cmp::Ordering;

        #[derive(Debug, Eq, PartialEq)]
        struct State {
            cost: u64,
            node: Uuid,
        }

        impl Ord for State {
            fn cmp(&self, other: &Self) -> Ordering {
                other.cost.cmp(&self.cost)
            }
        }
        impl PartialOrd for State {
            fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
                Some(self.cmp(other))
            }
        }

        let mut dist: HashMap<Uuid, u64> = HashMap::new();
        let mut prev: HashMap<Uuid, Option<Uuid>> = HashMap::new();
        let mut heap = BinaryHeap::new();

        dist.insert(source, 0);
        heap.push(State { cost: 0, node: source });

        while let Some(State { cost, node }) = heap.pop() {
            if node == target {
                break;
            }

            if cost > *dist.get(&node).unwrap_or(&u64::MAX) {
                continue;
            }

            for edge in &graph.edges {
                if edge.from == node {
                    let next = edge.to;
                    let next_cost = cost + edge.latency_ms;

                    if next_cost < *dist.get(&next).unwrap_or(&u64::MAX) {
                        dist.insert(next, next_cost);
                        prev.insert(next, Some(node));
                        heap.push(State { cost: next_cost, node: next });
                    }
                }
            }
        }

        // Reconstruct path
        let mut path = Vec::new();
        let mut current = target;
        
        if !prev.contains_key(&target) && source != target {
            return Err("No path found".to_string());
        }

        path.push(current);
        while let Some(Some(previous)) = prev.get(&current) {
            path.push(*previous);
            current = *previous;
        }
        path.reverse();

        Ok(path)
    }

    pub async fn update_routing_weights(&self, feedback: RoutingFeedback) {
        let mut weights = self.routing_weights.write().await;
        
        // Adaptive weight adjustment based on feedback
        let learning_rate = 0.01;
        
        weights.latency_weight += learning_rate * (feedback.actual_latency - feedback.predicted_latency);
        weights.fee_weight += learning_rate * (feedback.actual_fee - feedback.predicted_fee);
        
        // Normalize weights
        let total = weights.latency_weight + weights.fee_weight + 
                   weights.security_weight + weights.liquidity_weight + 
                   weights.reliability_weight;
        
        weights.latency_weight /= total;
        weights.fee_weight /= total;
        weights.security_weight /= total;
        weights.liquidity_weight /= total;
        weights.reliability_weight /= total;
    }
}

pub struct TrustEngine {
    trust_scores: Arc<RwLock<HashMap<Uuid, TrustVector>>>,
    behavior_profiles: Arc<RwLock<HashMap<Uuid, BehaviorProfile>>>,
    identity_mesh: Arc<IdentityMesh>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BehaviorProfile {
    pub entity_id: Uuid,
    pub avg_transaction_value: f64,
    pub transaction_frequency: f64,
    pub typical_hours: Vec<u32>,
    pub preferred_regions: Vec<String>,
    pub device_fingerprints: Vec<String>,
    pub risk_tolerance: f64,
}

impl TrustEngine {
    pub fn new(identity_mesh: Arc<IdentityMesh>) -> Self {
        TrustEngine {
            trust_scores: Arc::new(RwLock::new(HashMap::new())),
            behavior_profiles: Arc::new(RwLock::new(HashMap::new())),
            identity_mesh,
        }
    }

    pub async fn compute_trust(
        &self,
        entity: &EntityDescriptor,
        transaction: &HelixTransaction,
    ) -> TrustVector {
        // Identity confidence from identity mesh
        let identity_confidence = self.identity_mesh
            .verify_identity(&entity.id)
            .await;

        // Behavioral consistency
        let behavioral_consistency = self
            .analyze_behavioral_consistency(entity, transaction)
            .await;

        // Transaction integrity
        let transactional_integrity = self
            .analyze_transaction_integrity(transaction)
            .await;

        // Network reputation
        let network_reputation = self
            .compute_network_reputation(entity)
            .await;

        // Fraud exposure
        let fraud_exposure = self
            .assess_fraud_exposure(entity, transaction)
            .await;

        let trust_vector = TrustVector {
            identity_confidence,
            behavioral_consistency,
            transactional_integrity,
            network_reputation,
            fraud_exposure,
        };

        // Store updated trust score
        self.trust_scores
            .write()
            .await
            .insert(entity.id, trust_vector.clone());

        trust_vector
    }

    async fn analyze_behavioral_consistency(
        &self,
        entity: &EntityDescriptor,
        transaction: &HelixTransaction,
    ) -> f64 {
        let profiles = self.behavior_profiles.read().await;
        
        if let Some(profile) = profiles.get(&entity.id) {
            let mut consistency = 1.0;
            
            // Check transaction value anomaly
            if transaction.value.value > profile.avg_transaction_value * 3.0 {
                consistency -= 0.2;
            }
            
            // Check time anomaly
            let current_hour = transaction.timestamp.hour();
            if !profile.typical_hours.contains(&current_hour) {
                consistency -= 0.1;
            }
            
            consistency.max(0.0)
        } else {
            0.5 // Default for unknown entities
        }
    }

    async fn analyze_transaction_integrity(
        &self,
        transaction: &HelixTransaction,
    ) -> f64 {
        // Analyze transaction structure
        let mut integrity = 1.0;
        
        if transaction.value.value <= 0.0 {
            integrity = 0.0;
        }
        
        if transaction.source.id == transaction.destination.id {
            integrity -= 0.5;
        }
        
        integrity.max(0.0)
    }

    async fn compute_network_reputation(
        &self,
        entity: &EntityDescriptor,
    ) -> f64 {
        let scores = self.trust_scores.read().await;
        
        // PageRank-like algorithm for reputation
        let mut reputation = 0.5;
        
        if let Some(trust) = scores.get(&entity.id) {
            reputation = trust.network_reputation;
        }
        
        // Decay factor
        reputation *= 0.99;
        
        reputation
    }

    async fn assess_fraud_exposure(
        &self,
        entity: &EntityDescriptor,
        transaction: &HelixTransaction,
    ) -> f64 {
        // Fraud exposure calculation
        let mut exposure = 0.0;
        
        // High-value transactions increase exposure
        if transaction.value.value > 10000.0 {
            exposure += 0.3;
        }
        
        // Unknown entities increase exposure
        if entity.trust_level < 0.3 {
            exposure += 0.4;
        }
        
        exposure.min(1.0)
    }
}

pub struct IdentityMesh {
    identities: Arc<RwLock<HashMap<Uuid, IdentityRecord>>>,
    credentials: Arc<RwLock<HashMap<Uuid, Vec<VerifiableCredential>>>>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct IdentityRecord {
    pub id: Uuid,
    pub did: String, // Decentralized Identifier
    pub public_key: Vec<u8>,
    pub created_at: DateTime<Utc>,
    pub verified: bool,
    pub verification_methods: Vec<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct VerifiableCredential {
    pub id: Uuid,
    pub issuer: String,
    pub subject: Uuid,
    pub credential_type: String,
    pub claims: HashMap<String, String>,
    pub issued_at: DateTime<Utc>,
    pub expires_at: Option<DateTime<Utc>>,
    pub signature: Vec<u8>,
}

impl IdentityMesh {
    pub fn new() -> Self {
        IdentityMesh {
            identities: Arc::new(RwLock::new(HashMap::new())),
            credentials: Arc::new(RwLock::new(HashMap::new())),
        }
    }

    pub async fn register_identity(
        &self,
        did: String,
        public_key: Vec<u8>,
    ) -> Result<Uuid, String> {
        let id = Uuid::new_v4();
        
        let record = IdentityRecord {
            id,
            did,
            public_key,
            created_at: Utc::now(),
            verified: false,
            verification_methods: vec!["ed25519".to_string()],
        };
        
        self.identities.write().await.insert(id, record);
        Ok(id)
    }

    pub async fn verify_identity(&self, entity_id: &Uuid) -> f64 {
        let identities = self.identities.read().await;
        let credentials = self.credentials.read().await;
        
        if let Some(identity) = identities.get(entity_id) {
            let mut confidence = 0.2; // Base score for having an identity
            
            if identity.verified {
                confidence += 0.3;
            }
            
            // Check for valid credentials
            if let Some(creds) = credentials.get(entity_id) {
                confidence += (creds.len() as f64 * 0.1).min(0.5);
            }
            
            confidence.min(1.0)
        } else {
            0.0
        }
    }

    pub async fn issue_credential(
        &self,
        issuer: String,
        subject: Uuid,
        credential_type: String,
        claims: HashMap<String, String>,
    ) -> Result<Uuid, String> {
        let id = Uuid::new_v4();
        
        let credential = VerifiableCredential {
            id,
            issuer,
            subject,
            credential_type,
            claims,
            issued_at: Utc::now(),
            expires_at: Some(Utc::now() + chrono::Duration::days(365)),
            signature: vec![],
        };
        
        self.credentials
            .write()
            .await
            .entry(subject)
            .or_insert_with(Vec::new)
            .push(credential);
        
        Ok(id)
    }
}

struct BFTEngine {
    quorum_size: usize,
    consensus_threshold: f64,
}

impl BFTEngine {
    fn new() -> Self {
        BFTEngine {
            quorum_size: 67, // 2/3 majority
            consensus_threshold: 0.67,
        }
    }

    async fn reach_consensus(
        &self,
        validators: &[ValidatorNode],
        transaction: &HelixTransaction,
    ) -> bool {
        // Sort validators by reputation * stake
        let mut sorted_validators = validators.to_vec();
        sorted_validators.sort_by(|a, b| {
            (b.reputation * b.stake)
                .partial_cmp(&(a.reputation * a.stake))
                .unwrap()
        });

        let total_votes = sorted_validators.len() as f64;
        let mut approval_votes = 0.0;

        for validator in &sorted_validators {
            // Validator votes with weight proportional to reputation * stake
            if self.validate_transaction(validator, transaction).await {
                approval_votes += validator.reputation * validator.stake;
            }
        }

        (approval_votes / total_votes) >= self.consensus_threshold
    }

    async fn validate_transaction(
        &self,
        validator: &ValidatorNode,
        transaction: &HelixTransaction,
    ) -> bool {
        // Complex validation logic
        transaction.value.value > 0.0 
        && transaction.source.id != transaction.destination.id
        && validator.reputation > 0.5
    }
}

struct ProbabilisticEngine {
    confidence_threshold: f64,
}

impl ProbabilisticEngine {
    fn new() -> Self {
        ProbabilisticEngine {
            confidence_threshold: 0.95,
        }
    }

    async fn probabilistic_consensus(
        &self,
        nodes: &[ValidatorNode],
        transaction: &HelixTransaction,
    ) -> (bool, f64) {
        let mut confidence = 0.0;
        let mut successful_validations = 0;

        for _ in 0..100 {
            let random_validator = &nodes[rand::thread_rng().gen_range(0..nodes.len())];
            
            if self.quick_validate(random_validator, transaction).await {
                successful_validations += 1;
            }
        }

        confidence = successful_validations as f64 / 100.0;
        (confidence >= self.confidence_threshold, confidence)
    }

    async fn quick_validate(
        &self,
        validator: &ValidatorNode,
        transaction: &HelixTransaction,
    ) -> bool {
        transaction.fraud_probability < 0.3 
        && validator.performance > 0.7
    }
}

impl HybridConsensus {
    pub fn new() -> Self {
        HybridConsensus {
            validators: Arc::new(RwLock::new(Vec::new())),
            current_mode: Arc::new(RwLock::new(ConsensusMode::ProbabilisticFederated)),
            bft_engine: BFTEngine::new(),
            probabilistic_engine: ProbabilisticEngine::new(),
        }
    }

    pub async fn select_consensus_mode(
        &self,
        transaction: &HelixTransaction,
        network_state: &NetworkState,
    ) -> ConsensusMode {
        if transaction.value.value > 1_000_000.0 {
            return ConsensusMode::DeterministicBFT;
        }

        if network_state.threat_level > 0.8 {
            return ConsensusMode::DeterministicBFT;
        }

        if network_state.congestion > 0.7 {
            return ConsensusMode::OptimisticExecution;
        }

        if transaction.value.value < 100.0 {
            return ConsensusMode::EdgeSynchronized;
        }

        ConsensusMode::ProbabilisticFederated
    }

    pub async fn execute_consensus(
        &self,
        transaction: &HelixTransaction,
    ) -> Result<ConsensusResult, String> {
        let mode = self.current_mode.read().await.clone();
        let validators = self.validators.read().await;

        match mode {
            ConsensusMode::DeterministicBFT => {
                let approved = self.bft_engine
                    .reach_consensus(&validators, transaction)
                    .await;
                
                Ok(ConsensusResult {
                    approved,
                    confidence: if approved { 1.0 } else { 0.0 },
                    mode,
                    latency_ms: 2000,
                })
            }
            ConsensusMode::ProbabilisticFederated => {
                let (approved, confidence) = self.probabilistic_engine
                    .probabilistic_consensus(&validators, transaction)
                    .await;
                
                Ok(ConsensusResult {
                    approved,
                    confidence,
                    mode,
                    latency_ms: 500,
                })
            }
            ConsensusMode::OptimisticExecution => {
                Ok(ConsensusResult {
                    approved: true,
                    confidence: 0.90,
                    mode,
                    latency_ms: 50,
                })
            }
            ConsensusMode::EdgeSynchronized => {
                Ok(ConsensusResult {
                    approved: true,
                    confidence: 0.85,
                    mode,
                    latency_ms: 5,
                })
            }
        }
    }
}

    pub features: Vec<String>,
    pub weight: f64,
}

impl FraudNeuralEngine {
    pub fn new() -> Self {
        FraudNeuralEngine {
            fraud_threshold: 0.7,
            anomaly_patterns: Arc::new(RwLock::new(Vec::new())),
        }
    }

    pub async fn analyze_transaction(
        &self,
        transaction: &HelixTransaction,
        user_history: &[HelixTransaction],
    ) -> f64 {
        let mut fraud_score = 0.0;

        // Check transaction value anomaly
        if !user_history.is_empty() {
            let avg_value: f64 = user_history.iter()
                .map(|tx| tx.value.value)
                .sum::<f64>() / user_history.len() as f64;
            
            if transaction.value.value > avg_value * 5.0 {
                fraud_score += 0.3;
            }
        }

        // Check velocity (multiple transactions in short time)
        let recent_txs = user_history.iter()
            .filter(|tx| {
                (transaction.timestamp - tx.timestamp).num_seconds() < 60
            })
            .count();
        
        if recent_txs > 10 {
            fraud_score += 0.4;
        }

        // Check for known fraud patterns
        let patterns = self.anomaly_patterns.read().await;
        for pattern in patterns.iter() {
            if self.match_pattern(transaction, pattern) {
                fraud_score += pattern.weight;
            }
        }

        fraud_score.min(1.0)
    }

    fn match_pattern(&self, tx: &HelixTransaction, pattern: &FraudPattern) -> bool {
        for feature in &pattern.features {
            match feature.as_str() {
                "high_value" => {
                    if tx.value.value < 10000.0 {
                        return false;
                    }
                }
                "new_account" => {
                    if tx.source.trust_level > 0.5 {
                        return false;
                    }
                }
                _ => {}
            }
        }
        true
    }

    pub async fn should_block(&self, fraud_score: f64) -> bool {
        fraud_score > self.fraud_threshold
    }
}

            .max_by(|a, b| {
                a.yield_rate.partial_cmp(&b.yield_rate).unwrap()
            })
            .cloned()
    }

        &self,

                if pool.utilization > self.rebalancing_threshold {
                    // Trigger rebalancing
                    self.rebalance_pool(pool_id).await?;
                }
                
                Ok(amount)
            } else {
                Err("Insufficient liquidity".to_string())
            }

}

impl HelixProtocol {
    pub fn new() -> Self {
        let ledger = Arc::new(AdaptiveLedger::new());
        let identity_mesh = Arc::new(IdentityMesh::new());
        let aer_engine = Arc::new(AEREngine::new());
        let trust_engine = Arc::new(TrustEngine::new(identity_mesh.clone()));
        let consensus_engine = Arc::new(HybridConsensus::new());
        let fraud_engine = Arc::new(FraudNeuralEngine::new());
        let liquidity_engine = Arc::new(LiquidityEngine::new());
        let state_engine = Arc::new(UniversalStateEngine::new(ledger.clone()));

        HelixProtocol {
            ledger,
            aer_engine,
            trust_engine,
            consensus_engine,
            fraud_engine,
            liquidity_engine,
            state_engine,
            identity_mesh,
        }
    }

    pub async fn process_transaction(
        &self,
        source: EntityDescriptor,
        destination: EntityDescriptor,
        value: Amount,
    ) -> Result<HelixTransaction, String> {
        // 1. Create transaction
        let mut tx = HelixTransaction {
            id: Uuid::new_v4(),
            timestamp: Utc::now(),
            source: source.clone(),
            destination: destination.clone(),
            value: value.clone(),
            asset_class: AssetType::Fiat("USD".to_string()),
            route: AdaptiveRoute {
                hops: vec![],
                chosen_by: RouteDecision::Optimal,
                estimated_latency_ms: 0,
                estimated_fee: 0.0,
                liquidity_depth: 0.0,
                security_level: SecurityLevel::High,
                fallback_routes: vec![],
                policies: vec![],
            },
            trust_score: TrustVector {
                identity_confidence: 0.0,
                behavioral_consistency: 0.0,
                transactional_integrity: 0.0,
                network_reputation: 0.0,
                fraud_exposure: 0.0,
            },
            fraud_probability: 0.0,
            status: TransactionStatus::Initiated,
            policies: vec![],
        };

        // 2. Compute trust
        tx.trust_score = self.trust_engine.compute_trust(&source, &tx).await;
        
        if tx.trust_score.identity_confidence < 0.3 {
            return Err("Identity verification failed".to_string());
        }

        // 3. Check fraud
        tx.fraud_probability = self.fraud_engine.analyze_transaction(&tx, &[]).await;
        
        if self.fraud_engine.should_block(tx.fraud_probability).await {
            tx.status = TransactionStatus::Failed("Fraud detected".to_string());
            return Err("Transaction blocked by fraud engine".to_string());
        }

        // 4. Compute optimal route
        tx.route = self.aer_engine.compute_optimal_route(
            &source,
            &destination,
            &value,
        ).await?;
        
        tx.status = TransactionStatus::Routing;

        // 5. Allocate liquidity
        if let Some(pool_id) = tx.route.hops.first().map(|h| h.node_id.to_string()) {
            self.liquidity_engine.allocate_liquidity(&pool_id, value.value).await?;
        }

        // 6. Execute consensus
        let consensus_result = self.consensus_engine.execute_consensus(&tx).await?;
        
        if !consensus_result.approved {
            tx.status = TransactionStatus::Failed("Consensus failed".to_string());
            return Err("Consensus not reached".to_string());
        }

        // 7. Record in ledger
        tx.status = TransactionStatus::Settling;
        self.ledger.record_transaction(tx.clone()).await?;
        
        // 8. Update state
        self.state_engine.synchronize_state().await?;
        
        // 9. Complete
        tx.status = TransactionStatus::Completed;
        
        Ok(tx)
    }
}

const resolvers = {
  Query: {
    me: async (_: any, __: any, context: any) => {
      if (!context.user) throw new Error('Not authenticated');
      return context.user;
    },
    networkStatus: async () => {
      const cached = await redisClient.get('network:status');
      return cached ? JSON.parse(cached) : {};
    },
    liquidityPools: async () => {
      const cached = await redisClient.get('network:liquidity');
      return cached ? JSON.parse(cached) : [];
    },
  },
  Mutation: {
    sendTransaction: async (_: any, args: any, context: any) => {
      if (!context.user) throw new Error('Not authenticated');
      
      const transaction = {
        id: uuidv4(),
        source: { id: context.user.id },
        destination: { id: args.destination },
        amount: args.amount,
        currency: args.currency || 'USDC',
        status: 'INITIATED',
        timestamp: new Date().toISOString(),
      };
      
      await redisClient.publish('transactions', JSON.stringify(transaction));
      return transaction;
    },
  },
};

    if (token) {

        const user = jwt.verify(token, process.env.JWT_SECRET || 'helix-secret');

        return;
      }
    }
    
    wss.handleUpgrade(request, socket, head, (ws) => {
      wss.emit('connection', ws, request);
    });
  });

        require(balanceOf[msg.sender] >= _value, "Insufficient balance");
        
        balanceOf[msg.sender] -= _value;
        balanceOf[_to] += _value;
        
        emit Transfer(msg.sender, _to, _value);
        return true;
    }
    
    function approve(address _spender, uint256 _value) public returns (bool) {
        allowance[msg.sender][_spender] = _value;
        emit Approval(msg.sender, _spender, _value);
        return true;
    }
    
    function transferFrom(address _from, address _to, uint256 _value) public returns (bool) {
        require(balanceOf[_from] >= _value, "Insufficient balance");
        require(allowance[_from][msg.sender] >= _value, "Allowance exceeded");
        
        balanceOf[_from] -= _value;
        balanceOf[_to] += _value;
        allowance[_from][msg.sender] -= _value;
        
        emit Transfer(_from, _to, _value);
        return true;
    }
